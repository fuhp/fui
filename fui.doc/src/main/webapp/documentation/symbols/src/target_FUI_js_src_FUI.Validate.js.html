222         FValidate : function(options) {
223 
224             // if nothing is selected, return nothing; can't chain anyway
225             if (!this.length) {
226                 options && options.debug && window.console && console.warn("nothing selected, can't validate, returning nothing");
227 
228                 return;
229             }
230 
231             // check if a validator for this form was already created
232             var validator = $.data(this[0], 'validator');
233             if (validator) {
234                 return validator;
235             }
236 
237             // Add novalidate tag if HTML5.
238             this.attr('novalidate', 'novalidate');
239             //特殊处理onkeyup属性,如果用户设置true，则使用默认的onkeyup函数。即只接收false属性。
240             if (options.onkeyup) {
241                 delete options.onkeyup;
242             }
243             //特殊处理onclick属性，如果用户设置true，则使用默认的onclick函数。即只接收false属性。
244             if (options.onclick) {
245                 delete options.onclick;
246             }
247             //特殊处理onfocusout属性，如果用户设置true，则使用默认的onfocusout函数。即只接收false属性。
248             if (options.onfocusout) {
249                 delete options.onfocusout;
250             }
251             //特殊处理onsubmit属性，如果用户设置true，则使用默认的onsubmit函数。即只接收false属性。
252             if (options.onsubmit) {
253                 delete options.onsubmit;
254             }
255             /**
256              * 如果用户设置了errorModel，且值为qtip或者under，则执行系统默认制定的方式。
257              * 否则，如果用户自己制定显示方式，设置errorPlacement属性，则按照用户自己设置的方式进行。
258              */
259             if (!options.errorPlacement) {
260                 if ("qtip" === options.errorModel) {
261                     options.errorMsgClass = "f-validate-errorMsg";
262                     options.errorElement = "label";
263                     options.errorPlacement = function(error, element) {
264                         if (error.html()) {
265                             var el = $(element);
266                             //var error = error.wrap("<span id='fui-validate-qtip-msg' class='f-validate-errorMsg' style='display:none;' />").parent();
267 
268                             var qtipMsgEl = $('#fui-validate-qtip-msg');
269                             if (!qtipMsgEl.length) {
270                                 $('body').append("<span id='fui-validate-qtip-msg' class='f-validate-errorMsg' style='display:none;' />");
271                             }
272 
273 
274                             var errorMsgTarget = el.attr("errorMsgTarget");
275                             //例如combo number 组件有两个input
276                             if (errorMsgTarget) {
277                                 if ("parent" == errorMsgTarget) {
278                                     //制定节点查找其兄弟节点
279                                     var parent = el.parent();
280                                     parent.attr('f-validate-qtip-msg', error.html());
281                                 } else {
282                                     //通过id来查找
283                                     errorMsgTarget = $.trim(errorMsgTarget);
284                                     if (errorMsgTarget.indexOf("#") !== 0) {
285                                         errorMsgTarget = "#" + errorMsgTarget;
286                                     }
287                                     var parent = $(errorMsgTarget);
288                                     parent.attr('f-validate-qtip-msg', error.html());
289                                 }
290                             } else {
291                                 el.attr('f-validate-qtip-msg', error.html());
292                             }
293                         }
294                     };
295                     options.showErrors = function(errorMap, errorList) {
296                         if (errorList && errorList.length > 0) {
297                             $.each(errorList, function(index, obj) {
298                                 var msg = this.message;
299                                 var el = $(obj.element);
300                                 var errorMsgTarget = el.attr("errorMsgTarget");
301                                 //绑定一个次事件
302                                 if (el.attr("hasbindmouseevent") !== "true") {
303                                     if (errorMsgTarget) {
304                                         var parentEl = null;
305                                         if ("parent" == errorMsgTarget) {
306                                             //指定父节点查找其兄弟节点
307                                             parentEl = el.parent();
308                                         } else {
309                                             //通过id来查找
310                                             errorMsgTarget = $.trim(errorMsgTarget);
311                                             if (errorMsgTarget.indexOf("#") !== 0) {
312                                                 errorMsgTarget = "#" + errorMsgTarget;
313                                             }
314                                             parentEl = $(errorMsgTarget);
315                                         }
316                                         //判断是否有错误提示消息
317                                         //if (msgEl.html().length > 0 && msgEl.attr("class").length > 0 && msgEl.find("label").html().length > 0) {
318                                         parentEl.bind('mouseover',
319                                                 function(e) {
320                                                     var qtipMsgEl = $('#fui-validate-qtip-msg');
321                                                     var el = $(this);
322                                                     var msg = el.attr('f-validate-qtip-msg');
323                                                     var valid =  el.attr('valid');
324                                                     if (valid !== 'true' &&qtipMsgEl.length) {
325                                                         qtipMsgEl.html(msg);
326                                                         qtipMsgEl.css('display', 'inline').css({'top':e.pageY + 10 , 'left':e.pageX + 5});
327                                                     }
328                                                 }).bind('mouseout',
329                                                 function() {
330                                                     var qtipMsgEl = $('#fui-validate-qtip-msg');
331                                                     if (qtipMsgEl.length) {
332                                                         qtipMsgEl.css('display', 'none');
333                                                     }
334                                                 });
335 
336                                     } else {
337                                         //直接綁定
338                                         el.bind('mouseover',
339                                                 function(e) {
340                                                     var qtipMsgEl = $('#fui-validate-qtip-msg');
341                                                     var el = $(this);
342                                                     var msg = el.attr('f-validate-qtip-msg');
343                                                     var valid =  el.attr('valid');
344                                                      if (valid !== 'true' &&qtipMsgEl.length) {
345                                                         qtipMsgEl.html(msg);
346                                                         qtipMsgEl.css('display', 'inline').css({'top':e.pageY + 10 , 'left':e.pageX + 5});
347                                                     }
348                                                 }).bind('mouseout',
349                                                 function() {
350                                                     var qtipMsgEl = $('#fui-validate-qtip-msg');
351                                                     if (qtipMsgEl.length) {
352                                                         qtipMsgEl.css('display', 'none');
353                                                     }
354                                                 });
355                                     }
356                                     el.attr("hasbindmouseevent", true);
357                                 }
358                             });
359                         } else {
360                             $(this.currentElements).parents().map(function() {
361                                 if (this.tagName.toLowerCase() == 'td') {
362                                     //$(this).children().eq(1).hide();
363                                     $(this).children().eq(0).removeClass("error-border");
364                                 } else {
365                                     $(this).removeClass("error-border");
366                                 }
367                                 $(this).children().eq(0).removeClass("x-form-invalid");
368                             });
369                         }
370                         this.defaultShowErrors();
371                     }
372                 } else {
373                     //用户没有设置errorPlacement，errorModel使用默认值：under
374                     options.errorPlacement = function(error, element) {
375                         if (error.html()) {
376                             var el = $(element);
377                             var errorMsgTarget = el.attr("errorMsgTarget");
378                             //例如combo number 组件有两个input
379                             if (errorMsgTarget) {
380                                 if ("parent" == errorMsgTarget) {
381                                     //制定节点查找其兄弟节点
382                                     var parent = el.parent();
383                                     error.insertAfter(parent);
384                                     error.css('display', 'none');
385 
386                                 } else {
387                                     //通过id来查找
388                                     errorMsgTarget = $.trim(errorMsgTarget);
389                                     if (errorMsgTarget.indexOf("#") !== 0) {
390                                         errorMsgTarget = "#" + errorMsgTarget;
391                                     }
392                                     var parent = $(errorMsgTarget);
393                                     error.insertAfter(parent);
394                                     error.css('display', 'none');
395                                 }
396                             } else {
397                                 //纯的html元素，例如input
398                                 error.insertAfter(el);
399                                 error.css('display', 'none');
400                             }
401                         }
402                     };
403                 }
404 
405             }
406 
407 
408             validator = new $.fvalidator(options, this[0]);
409             $.data(this[0], 'validator', validator);
410 
411             if (validator.settings.onsubmit) {
412 
413                 //var inputsAndButtons = this.find("input, button");
414                 var inputs = this.find('input');
415                 var buttons = this.find('button');
416 
417                 var inputsAndButtons = inputs.add(buttons);
418 
419                 // allow suppresing validation by adding a cancel class to the submit button
420                 inputsAndButtons.filter(".cancel").click(function () {
421                     validator.cancelSubmit = true;
422                 });
423 
424                 // when a submitHandler is used, capture the submitting button
425                 if (validator.settings.submitHandler) {
426                     inputsAndButtons.filter(":submit").click(function () {
427                         validator.submitButton = this;
428                     });
429                 }
430 
431                 // validate the form on submit
432                 this.submit(function(event) {
433                     if (validator.settings.debug)
434                     // prevent form submit to be able to see console output
435                         event.preventDefault();
436 
437                     function handle() {
438                         if (validator.settings.submitHandler) {
439                             if (validator.submitButton) {
440                                 // insert a hidden input as a replacement for the missing submit button
441                                 var hidden = $("<input type='hidden'/>").attr("name", validator.submitButton.name).val(validator.submitButton.value).appendTo(validator.currentForm);
442                             }
443                             validator.settings.submitHandler.call(validator, validator.currentForm);
444                             if (validator.submitButton) {
445                                 // and clean up afterwards; thanks to no-block-scope, hidden can be referenced
446                                 hidden.remove();
447                             }
448                             return false;
449                         }
450                         return true;
451                     }
452 
453                     // prevent submit for invalid forms or custom submit handlers
454                     if (validator.cancelSubmit) {
455                         validator.cancelSubmit = false;
456                         return handle();
457                     }
458                     if (validator.form()) {
459                         if (validator.pendingRequest) {
460                             validator.formSubmitted = true;
461                             return false;
462                         }
463                         return handle();
464                     } else {
465                         validator.focusInvalid();
466                         return false;
467                     }
468                 });
469             }
470             //validator.checkForm();
471             return validator;
472         }
473 
474         //校验单个控件
475         ,isValidate : function(formId) {
476             if (formId) {
477                 var form = $I(formId);
478                 if (form.length > 0) {
479                     var valid = false;
480                     var validator = form.validate();
481                     valid = validator.element(this);
482                     return valid;
483                 } else {
484                     return true;
485                 }
486             } else {
487                 var valid = false;
488                 var element = this[0];
489                 if (element.form) {
490                     var validator = $.data(element.form, 'validator');
491                     if (validator) {
492                         valid = validator.element(this);
493                         return valid;
494                     } else {
495                         return true;
496                     }
497                 } else {
498                     return true;
499                 }
500             }
501         }
502 
503     });
