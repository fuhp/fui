 15 (function($) {
 16 	$.registerWidgetEvent("");
 17 	// 使回车键即tab键：保证容器内部的有效的input组件，在点击enter键时，按照tabIndex顺序切换到下一个，并保证在容器内部循环
 18 	$.widget("FUI.FEnterSwitch", {
 19 	    options : {
 20 	    },
 21 	    _create : function() {
 22 	    	this.options.id = this.element.attr("id");
 23 	    },
 24 	    _init : function() {
 25 	    	this._bindEvent();
 26 	    },
 27 	    
 28 	    _bindEvent : function() {
 29 	    	var self = this;
 30 	    	self._getAllEls().keyup(function(e) {
 31 	    		if (e.which == 13 || e.which == 108) { // 判断是否是回车键或者小键盘回车键
 32 	    			// TODO 性能有待优化！
 33 	    			// 实时获取所有可见有效的表单元素，允许用户在运行时将某些组件隐藏然后显示出来
 34 	    			var MAX_GAP = 500;
 35 	    			var els = self._getAllVisible().get();
 36 	    			var curIndex = $.inArray(this, els);
 37 	    			var mine = parseInt(this.tabIndex || 0) * MAX_GAP + curIndex;
 38 	    	    	var minGap = 10000000; // 保存最小的差距，初始值很大
 39 	    	    	var minGapIndex = -1; // 最小差距的元素索引，即下一个元素的索引
 40 	    	    	var min = mine; // tabIndex最小的值
 41 	    	    	var minIndex = curIndex; // 拥有tabIndex最小值 的元素的索引
 42 	    	    	for (var i=0; i<els.length; i++) {
 43 	    	    		var other = parseInt(els[i].tabIndex || 0) * MAX_GAP + i;
 44 	    	    		if (other < min) {
 45 	    	    			min = other;
 46 	    	    			minIndex = i;
 47 	    	    		}
 48 	    	    		if (curIndex != i) {
 49 	    	    			var curGap = other - mine;
 50 	    	    			if (curGap == 1) {
 51 	    	    				minGapIndex = i;
 52 	    	    				break;
 53 	    	    			}  else if (curGap > 0 && curGap < minGap) {
 54 	    	    				minGap = curGap;
 55 	    	    				minGapIndex = i;
 56 	    	    			}
 57 	    	    		}
 58 	    	    	}
 59 	    	    	var nextEl = null;
 60 	    	    	if (minGapIndex != -1) { // 存在下一个元素
 61 	    	    		nextEl = els[minGapIndex];
 62 	    	    	} else { // 如果已经是最大的tabIndex，则自动切换到最小的tabIndex
 63 	    	    		nextEl = els[minIndex];
 64 	    	    	}
 65     				self._focusEl($(nextEl));
 66 	    			return false; // 必须return false，否则会造成回车事件被传递给下一个组件比如textarea
 67 	    		}
 68 	    	});
 69 	    },
 70 	    
 71 	    _focusEl : function(element) {
 72 	    	var el = element;
 73 	    	// 必须使用setTimeout(,0) 否则在使用某些情况下会造成浏览器假死（比如my97datepicker在focus是弹出的情况下）
 74 	    	setTimeout(function() {  
 75 	    		el.focus();
 76 		    	if (el.is("input:text")) {
 77 		    		el.select();
 78 		    	}
 79 	    	},0)
 80 	    },
 81 	    
 82 	    _getAllEls : function() {
 83 	    	// 容器下所有的<input> <select> <button>元素，保证textarea可以被切换到，但是不能通过回车切换到下一个
 84 	    	return $(":input:not('textarea')", this.element);
 85 	    },
 86 	    
 87 	    _getAllVisible : function() {
 88 	    	// 所有可见的不是disabled的input、textarea、select的表单组件
 89 	    	return $(":input:enabled:visible", this.element);
 90 	    },
 91 	    
 92 	    /**
 93 	     * 销毁组件
 94 	     * @name FEnterSwitch#destroy
 95 	     * @function
 96 	     * @example
 97 	     */
 98 	    destroy : function() {
 99 		    var op = this.options;
100 		    // 调用父类的销毁方法
101 		    $.Widget.prototype.destroy.call(this);
102 	    }
103 	});
104 })(jQuery);
