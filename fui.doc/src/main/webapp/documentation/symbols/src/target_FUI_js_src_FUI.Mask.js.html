 27     function generateMaskHtml(text, options) {
 28         var id = options && options.id || 'fui' ;
 29         var maskBGId = id + '-mask-bg';
 30         var maskImgId = id + '-mask-img';
 31         var maskMsgId = id + '-mask-msg';
 32         var maskArr = [];
 33         //遮罩背景层
 34         maskArr.push('<div id="' + maskBGId + '"' + ' class="f-mask-bg"></div>');
 35         //遮罩层的提示图片以及提示信息
 36         maskArr.push('<div id="' + maskImgId + '" class="f-mask-img" style="display:none;"><div id="' + maskMsgId + '" class="">');
 37         maskArr.push(text || window["$i18n"].mask.defaultMsg);
 38         maskArr.push('</div></div>');
 39         return maskArr.join('');
 40     }

 46     function hiddenAllSelectEl() {
 47         if (window["hasHidden"] === true) {
 48             return;
 49         }
 50         var i = 0;
 51         var j = 0;
 52         var es;            //var es = vForm.elements;  //获取表单中所有的元素
 53         var hiddenArrayList = window['hiddenArrayList'] || [];
 54         window['hiddenArrayList'] = hiddenArrayList;
 55         window["hasHidden"] = true;
 56         var selectEls = $('select');
 57         var selectLen = selectEls.length;
 58         for (var i = 0; i < selectLen; i++) {
 59             var selectDom =selectEls.get(i);
 60             var selectStyle = selectDom.style ;
 61             if (selectStyle.display == "none" || selectStyle.display == "block") {
 62             } else {
 63                 selectStyle.display = "none";
 64                 hiddenArrayList.push(selectDom); //直接将对象的引用保存到链表中
 65             }
 66         }
 67     }

 69     //回复隐藏的select框
 70     function showAllSelectEl() {
 71         if (window["hasHidden"] === false) {
 72             return;
 73         }
 74         //todo 是否真的要隐藏select ，如果有多个遮罩层 处理需要谨慎
 75         var mask = $("#win-mask-overlay,#dialog-mask-overlay");
 76         if (mask.length > 0) {
 77             return;
 78         }
 79         window["hasHidden"] = false;
 80         var hiddenArrayList = window['hiddenArrayList'];
 81 
 82         var length = $.isArray(hiddenArrayList) && hiddenArrayList.length || 0;
 83         //将隐藏链表中的元素恢复显示
 84         if (hiddenArrayList != null && hiddenArrayList != false && length > 0) {
 85             for (var i = 0; i < length; i++) {
 86                 hiddenArrayList[i].style.display = "";   //继续隐藏
 87             }
 88         }
 89         hiddenArrayList = null;
 90         window['hiddenArrayList'] = null;
 91     }

141     //调整body区域的遮罩
142     // 20121213 hanyin 增加options参数传入
143     function _resizeBodyMask(text, maskId, options) {
144         //查找遮罩相关的元素
145         var maskBG = $('#' + maskId + '-mask-bg');
146         var maskImg = $('#' + maskId + '-mask-img');
147         var maskMsg = $('#' + maskId + '-mask-msg');
148 
149         var isBoxModel = jQuery.support.boxModel;
150         var documentElement = document.documentElement;
151 
152         //可视区域的宽度
153         var w = isBoxModel && documentElement.clientWidth ||document.body.clientWidth;
154         //可是区域的高度
155         var h = isBoxModel && documentElement.clientHeight ||document.body.clientHeight;
156         //滚动条的水平偏移量
157         var scrollLeft = isBoxModel && documentElement.scrollLeft || document.body.scrollLeft;
158         //滚动条的垂直偏移量
159         var scrollTop = isBoxModel && documentElement.scrollTop || document.body.scrollTop;
160 
161 
162         //设置背景阴影的宽高
163         var bw = (w + scrollLeft) + 'px';
164         var bh = (h + scrollTop) + 'px';
165         maskBG.css('width', bw);
166         maskBG.css('height', bh);
167 
168 
169         //设置消息提示信息内容
170         maskMsg.html(text);
171 
172         //遮罩图片的宽度
173         var imgW = maskImg.outerWidth(true);
174         //遮罩图片的高度
175         var imgH = maskImg.outerHeight(true);
176 
177         //设置遮罩图片的相对位置
178         var px = ((w - imgW) / 2 + scrollLeft) + 'px';
179         var py = ((h - imgH) / 2 + scrollTop) + 'px';
180         maskImg.css('left', px);
181         maskImg.css('top', py);
182 
183         maskBG.css('display', '');
184         //设置背景阴影和遮罩图片显现
185         // begin 20121213 hanyin 增加showImg 选项，如果为false，则不显示图片
186         if ((options || {})["showImg"] === false) {
187         	return;
188         }
189         // end 20121213 hanyin 增加showImg 选项，如果为false，则不显示图片
190         maskImg.css('display', '');
191     }

193     //调整div区域的遮罩
194     // 20121213 hanyin 增加options参数传入
195     function _resizeDivMask(text, maskId, options) {
196 
197         var element = $('#' + maskId);
198         //查找遮罩相关的元素
199         var maskBG = $('#' + maskId + '-mask-bg');
200         var maskImg = $('#' + maskId + '-mask-img');
201         var maskMsg = $('#' + maskId + '-mask-msg');
202 
203         //计算所需遮罩的宽度
204         var w = element.outerWidth(true);
205         //计算所需遮罩的高度
206         var h = element.outerHeight(true);
207         //滚动条的水平偏移量
208         var scrollLeft = element.scrollLeft();
209         //滚动条的垂直偏移量
210         var scrollTop = element.scrollTop();
211 
212 
213         //计算遮罩阴影的宽高
214         var bw = w + 'px';
215         var bh = h + 'px';
216         maskBG.css('width', bw);
217         maskBG.css('height', bh);
218 
219         //计算并设置遮罩阴影的相对偏移量。
220         var offset = element.offset();
221         var left = offset.left;
222         var top = offset.top;
223         maskBG.css('left', left);
224         maskBG.css('top', top);
225 
226         maskMsg.html(text);
227 
228         //计算遮罩图片的宽度
229         var imgW = maskImg.outerWidth(true);
230         //计算遮罩图片的高度
231         var imgH = maskImg.outerHeight(true);
232 
233         //计算并设置遮罩图片的偏移量
234         var px = (left + (w - imgW) / 2 ) + 'px';
235         var py = (top + (h - imgH) / 2 ) + 'px';
236         maskImg.css('left', px);
237         maskImg.css('top', py);
238 
239         maskBG.css('display', '');
240         // begin 20121213 hanyin 增加showImg 选项，如果为false，则不显示图片
241         if ((options || {})["showImg"] === false) {
242         	return;
243         }
244         // end 20121213 hanyin 增加showImg 选项，如果为false，则不显示图片
245         maskImg.css('display', '');
246     }

248     /**
249      * 遮罩某个指定的div
250      * @name FMask#doMask
251      * @param  text  遮罩提示信息（类型：String）
252      * @function
253      * @return void
254      * @example
255      */
256     // 20121213 hanyin _showBodyMask和_showDivMask方法的输入参数增加 options传入
257     $.fn.doMask = function(text, options) {
258         var element = $(this);
259         UTILS.addClass(element.get(0), 'f-mask-scroll-hide');
260         hideAllSelect();
261         //遮罩整个页面
262         if (element.is('body')) {
263             _showBodyMask(text, 'fui', options);
264         } else {
265             var id = element.attr('id');
266             _showDivMask(text, id, options);
267 
268         }
269     };

271     /**
272      * 去除某个指定div的遮罩
273      * @name FMask#doUnMask
274      * @function
275      * @return void
276      * @exampl
277      */
278     $.fn.doUnMask = function() {
279         var element = $(this);
280         showAllSelect();
281         //取消整个body区域的遮罩
282         if (element.is('body')) {
283             var maskBG = element.find('>#fui-mask-bg');
284             if (maskBG.length > 0) {
285                 var maskImg = element.find('>#fui-mask-img');
286                 maskImg.css('display', 'none');
287                 maskBG.css('display', 'none');
288             }
289             showAllSelect();
290         } else {
291             //取消制定div区域的遮罩
292             var id = element.attr('id');
293             var maskBGId = id + '-mask-bg';
294             var maskImgId = id + '-mask-img';
295             var maskBG = $('#' + maskBGId);
296             if (maskBG.length > 0) {
297                 var maskImg = $('#' + maskImgId);
298                 maskImg.css('display', 'none');
299                 maskBG.css('display', 'none');
300             }
301             showAllSelect();
302         }
303         UTILS.removeClass(element.get(0), 'f-mask-scroll-hide');
304     };
