1 /**
  2  * @name FPanel
  3  * @class 
  4  * 面板容器，是一种面向用户界面构建应用程序最佳的单元块，一种特定功能和结构化组件。
  5  * 面板包含有底部和顶部的工具条，连同单独的头部、底部和body部分。它提供内建都得可展开和可闭合的行为，
  6  * 连同多个内建的可制定的行为的工具按钮。面板可简易地置入任意的容器或布局。
  7  */

 55 /**
 56  * 组件创建时是否显示收起工具按钮(位于头部右边)。(仅标签使用)

 63 (function($) {
 64 	$.registerWidgetEvent("onExpand,onCollapse");
 65 	$.widget("FUI.FPanel", {
 66 	    options : {
 67 	        /**
 68 	         * 此组件的整体宽度（包括border和padding，单位象素），默认为"auto"。<br/>
 69 	         * 支持数字形式的字符串或者带px单位的数字字符串，比如"40"、"40px"，都表示40个像素，字符串如"px50"、"abcpx"等其他非法形式的字符串则被忽略；
 70 	         * width也支持百分比，比如"10%"，则表示占用外层容器(div,td等)宽度的比率，此时请慎重设置左右margin、border和padding，否则会造成在不同浏览器下表现不一致不准确。<br/>
 71 	         * @name FPanel#width
 72 	         * @type String
 73 	         * @default "auto"
 74 	         * @example
 75 	         * 无
 76 	         */
 77 	        width : "auto",
 78 	        /**
 79 	         * 此组件的整体高度（包括border和padding，单位像素），默认为auto。<br/>
 80 	         * 仅支持数字形式的字符串或者带px单位的数字字符串，比如"40"、"40px"，都是表示40个像素；其他非法形式的字符串则被忽略。<br/>
 81 	         * 请慎重设置组件的margin的top和bottom值，在不同浏览器下可能是不同的显示效果。
 82 	         * height属性不支持百分比。
 83 	         * @name FPanel#height
 84 	         * @type String
 85 	         * @default "auto"
 86 	         * @example
 87 	         * 无
 88 	         */
 89 	        height : "auto",
 90 	        /**
 91 	         * 异步加载内容的址
 92 	         * @name FPanel#pageUrl
 93 	         * @type String
 94 	         * @default ""
 95 	         * @example
 96 	         * 无
 97 	         */
 98 	        pageUrl : null,
 99 	        /**
100 	         * 是否采用iframe模式加载
101 	         * @name FPanel#isIframe
102 	         * @type boolean
103 	         * @default false
104 	         * @example
105 	         * 无
106 	         */
107 	        isIFrame : false,
108 	        /**
109 	         * panel的图标样式，位于头部左边的位置。(仅标签使用)
110 	         * @name FPanel#<ins>iconCls</ins>
111 	         * @type String
112 	         * @default ""
113 	         * @example
114 	         * 无
115 	         */
116 	        iconCls : null,
117 	        /**
118 	         * 收起panel组件后触发的函数
119 	         * @event
120 	         * @name FPanel#onCollapse
121 	         * @example
122 	         * onCollapse : function(){
123 	         *    //do something
124 	         * }
125 	         */
126 	        onCollapse : null,
127 	        /**
128 	         * 展开panel组件后触发的函数
129 	         * @event
130 	         * @name FPanel#onExpand
131 	         * @example
132 	         * onExpand : function(){
133 	         *    //do something
134 	         * }
135 	         */
136 	        onExpand : null,
137 	        /**
138 	         * 请求成功时触发
139 	         * @event
140 	         * @name FPanel#onLoadSuccess
141 	         * @param data  类型：Array[Object] 。请求返回的数据。
142 	         * @param textStatus   返回状态
143 	         * @param jqXHR   XMLHttpRequest对象
144 	         * @example
145 	         */
146 	        onLoadSuccess : null,
147 	        /**
148 	         * 请求超时时触发
149 	         * @event
150 	         * @name FPanel#onError
151 	         * @param jqXHR      XMLHttpRequest对象
152 	         * @param textStatus   返回状态
153 	         * @param  errorThrown  （可能）捕获的错误对象
154 	         * @example
155 	         */
156 	        onError : null,
157 
158 	        // 保存面板的当前状态：面板是否收起来
159 	        _isCollapsed : false,
160 	        // 缓存jQuery对象：header
161 	        _objHeader : null,
162 	        // 缓存jQuery对象：toggle
163 	        _objHeaderToggle : null,
164 	        // 缓存jQuery对象： title
165 	        _objHeaderTitle : null,
166 	        // 缓存jQuery对象： icon
167 	        _objHeaderIcon : null,
168 	        // 缓存jQuery对象：wrapper
169 	        _objWrapper : null,
170 	        // 缓存jQuery对象：body
171 	        _objWrapperBody : null
172 	    },
173 	    _create : function() {
174 		    // 初始化ID
175 		    var ID = this.element.attr("id");
176 		    this.options.id = ID;
177 		    // 初始化header对象
178 		    this.options._objHeader = $I(ID + "-header");
179 		    this.options._objHeaderToggle = $I(ID + "-toggle");
180 		    this.options._objHeaderTitle = $I(ID + "-title");
181 		    this.options._objHeaderIcon = $I(ID + "-icon");
182 		    // 初始化wrapper对象
183 		    this.options._objWrapper = $I(ID + "-wrapper");
184 		    this.options._objWrapperBody = $I(ID + "-body");
185 
186 		    // 是否折叠
187 		    this.options._isCollapsed = this.element.hasClass("f-collapsed");
188 		    // 绑定事件
189 		    this._bindEvent();
190 	    },
191 	    _init : function() {
192 		    // 如果是动态数据，并且开启自动加载，则加载数据
193 		    this.reload();
194 		    // 设置大小
195 		    this.setSize(this.options.width, this.options.height);
196 	    },
197 	    _bindEvent : function() {
198 		    var self = this;
199 		    var toggleEl = this.options._objHeaderToggle;
200 
201 		    // toggle绑定hover事件
202 		    toggleEl.hover(function(event) {
203 			    toggleEl.addClass("f-tool-toggle-over");
204 		    }, function(event) {
205 			    toggleEl.removeClass("f-tool-toggle-over");
206 		    });
207 
208 		    // 绑定点击事件
209 		    toggleEl.click(function() {
210 			    if (self.options._isCollapsed) {
211 				    self.expand();
212 			    } else {
213 				    self.collapse();
214 			    }
215 			    return false;
216 		    });
217 		    // 暂不支持
218 		    /*
219 		     * $(window).resize(function() { setTimeout(function() { self.resize(); }, 500); });
220 		     */
221 	    },
222 	    /**
223 	     * 销毁组件
224 	     * @name FPanel#destroy
225 	     * @function
226 	     * @example
227 	     */
228 	    destroy : function() {
229 		    var op = this.options;
230 		    // 取消绑定的事件
231 		    op._objHeaderToggle.unbind();
232 		    // 销毁jQuery对象
233 		    op._objHeader = null;
234 		    op._objHeaderToggle = null;
235 		    op._objHeaderTitle = null;
236 		    op._objHeaderIcon = null;
237 		    op._objWrapper = null;
238 		    op._objWrapperBody = null;
239 		    // 调用父类的销毁方法
240 		    $.Widget.prototype.destroy.call(this);
241 	    },
242 	    /**
243 	     * 重新加载指定的URL，如果传入的URL为空，则使用上一次使用的URL，如果没有配置URL，则直接返回，无任何效果
244 	     * @name FPanel#reload
245 	     * @function
246 	     * @param url 要加载的页面URL。在isFrame=true的情况下，url需要满足iframe的规范，在非iframe模式下，url不能跨域访问。
247 	     * @example
248 	     */
249 	    reload : function(url) {
250 		    var op = this.options;
251 		    var selfEl = this.element;
252 		    var bodyEl = op._objWrapperBody;
253 
254 		    url = url || op.pageUrl;
255 		    if ($Utils.isEmpty(url)) {
256 			    // 不是动态数据加载
257 			    return;
258 		    }
259 		    op.pageUrl = url;
260 
261 		    if (op.isIFrame) {
262 			    op._objWrapperBody.children("iframe").attr("src", url);
263 			    return;
264 		    }
265 
266 		    if (selfEl.data("loading")) {
267 			    // 如果在加载过程中，则直接跳出
268 			    return;
269 		    } else {
270 			    selfEl.data("loading", true);
271 		    }
272 		    if (url.indexOf("/") !== 0) {
273 			    url = $Utils.getContextPath() + '/' + url;
274 		    }
275 		    // TODO 此入显示MASK
276 		    $.ajax({
277 		        cache : false,
278 		        type : "get",
279 		        url : url,
280 		        success : function(data, textStatus, jqXHR) {
281 			        bodyEl.html(data);
282 			        if ($.isFunction(op.onLoadSuccess)) {
283 				        op.onLoadSuccess(data, textStatus, jqXHR);
284 			        }
285 		        },
286 		        error : function(data, textStatus, jqXHR) {
287 			        if ($.isFunction(op.onError)) {
288 				        op.onError(data, textStatus, jqXHR);
289 			        }
290 		        },
291 		        complete : function() {
292 			        selfEl.data("loading", false);
293 			        // TODO 隐藏MASK
294 		        }
295 		    });
296 	    },
297 	    /**
298 	     * 打开面板。
299 	     * @name FPanel#expand
300 	     * @function
301 	     * @example
302 	     */
303 	    expand : function() {
304 		    var self = this;
305 		    var op = self.options;
306 		    if (op._isCollapsed) {
307 			    op._objWrapper.slideDown("fast", function() {
308 				    self.element.addClass("f-collapsible").removeClass("f-collapsed");
309 				    op._isCollapsed = false;
310 				    // 调用回调或者触发事件
311 				    var canGo = true;
312 				    if ($.isFunction(op.onExpand)) {
313 					    var result = op.onExpand.call(self); // 回调方法返回false，则阻止触发事件
314 					    canGo = (result === false) ? false : true;
315 				    }
316 				    if (canGo) {
317 					    self.element.triggerHandler('onExpand');
318 				    }
319 			    });
320 		    }
321 	    },
322 	    /**
323 	     * 折叠面板。
324 	     * @name FPanel#collapse
325 	     * @function
326 	     * @example
327 	     */
328 	    collapse : function() {
329 		    var self = this;
330 		    var op = self.options;
331 		    if (!op._isCollapsed) {
332 			    op._objWrapper.slideUp("fast", function() {
333 				    self.element.removeClass("f-collapsible").addClass("f-collapsed");
334 				    op._isCollapsed = true;
335 				    // 调用回调或者触发事件
336 				    var canGo = true;
337 				    if ($.isFunction(op.onCollapse)) {
338 					    var result = op.onCollapse.call(self);
339 					    canGo = (result === false) ? false : true;
340 				    }
341 				    if (canGo) {
342 					    self.element.triggerHandler('onCollapse');
343 				    }
344 			    });
345 		    }
346 	    },
347 	    /**
348 	     * 设置标题。
349 	     * @name FPanel#setTitle
350 	     * @function
351 	     * @param title 标题
352 	     * @example
353 	     */
354 	    setTitle : function(title) {
355 		    if (title) {
356 			    this.options.title = title;
357 			    this.options._objHeaderTitle.text(title);
358 		    }
359 	    },
360 	    /**
361 	     * 设置标题图标样式。
362 	     * @name FPanel#setIconCls
363 	     * @function
364 	     * @param iconCls 图标样式
365 	     * @example
366 	     */
367 	    setIconCls : function(iconCls) {
368 		    var op = this.options;
369 		    // 移除原有样式
370 		    if (op.iconCls && op.iconCls.length != 0) {
371 			    op._objHeaderIcon.removeClass(op.iconCls);
372 		    }
373 		    // 设置新样式
374 		    op._objHeaderIcon.addClass(iconCls);
375 		    op.iconCls = iconCls;
376 	    },
377 	    /**
378 	     * 设置组件的高宽，Panel组件会自动调整内部toolbar的大小和位置
379 	     * @name FPanel#setSize
380 	     * @function
381 	     * @param w 组件的宽度，必须是数字或者数字类的字符串，不支持百分比
382 	     * @param h 组件的高度，必须是数字或者数字类的字符串，不支持百分比
383 	     * @example
384 	     */
385 	    setSize : function(w, h) {
386 		    var UTILS = window["$Utils"];
387 		    var op = this.options;
388 		    var selfEl = this.element;
389 		    var bodyEl = op._objWrapperBody;
390 		    var bodyContent = bodyEl.children(".f-panel-body-content");
391 		    var contentSize = bodyContent.size();
392 
393 		    if (w != null) {
394 			    // border、padding占用的大小
395 			    var percentage = UTILS.getPercentage(w);
396 			    if (percentage) {
397 				    selfEl.get(0).style.width = percentage;
398 				    bodyEl.get(0).style.width = "auto";
399 				    if (contentSize === 1) {
400 					    bodyContent.get(0).style.width = "auto";
401 				    }
402 			    } else {
403 				    // 对于'40'和'40px'等同样处理为40
404 				    var gaps = 2;
405 				    w = parseInt(w);
406 				    var realSize = w - gaps;
407 				    if (!isNaN(realSize)) {
408 					    if (realSize < 0) {
409 						    realSize = 0;
410 					    }
411 					    selfEl.get(0).style.width = w + "px";
412 					    if (contentSize === 1) {
413 						    bodyContent.get(0).style.width = realSize + "px";
414 					    } else {
415 						    bodyEl.get(0).style.width = realSize + "px";
416 					    }
417 				    }
418 			    }
419 		    }
420 		    if (h != null) {
421 			    // border、padding占用的大小
422 			    var percentage = UTILS.getPercentage(h);
423 			    if (percentage) {
424 				    selfEl.get(0).style.height = percentage;
425 				    bodyEl.get(0).style.height = "auto";
426 				    if (contentSize === 1) {
427 					    bodyContent.get(0).style.height = "auto";
428 				    }
429 			    } else {
430 				    // 对于'40'和'40px'等同样处理为40
431 				    var gaps = 27;
432 				    var realSize = parseInt(h) - gaps;
433 				    if (!isNaN(realSize)) {
434 					    if (realSize < 0) {
435 						    realSize = 0;
436 					    }
437 					    selfEl.get(0).style.height = "auto";
438 					    if (contentSize === 1) {
439 						    var otherHeight = 0;
440 						    bodyContent.siblings().not('script').each(function() {
441 							    // alert($(this).attr('id') + ":" + $(this).outerHeight(true));
442 							    otherHeight += $(this).outerHeight(true);
443 						    });
444 						    realSize = (realSize - otherHeight) < 0 ? 0 : (realSize - otherHeight);
445 						    bodyContent.get(0).style.height = realSize + "px";
446 					    } else {
447 						    bodyEl.get(0).style.height = realSize + "px";
448 					    }
449 				    }
450 			    }
451 		    }
452 	    }
453 	});
454 })(jQuery);
