 97 (function($) {
 98 	$.registerWidgetEvent("onValueChanged");
 99 	$.widget("FUI.FTextField", {
100 	    options : {
101 	    	/**
102 	    	 * 表单域的初始可用状态，如果为true则表示不可用，也不会参与表单提交；否则可以正常使用(仅标签使用)
103 	    	 * @name FTextField#<ins>disabled</ins>
104 	    	 * @type Boolean
105 	    	 * @default false
106 	    	 * @example
107 	    	 * 无
108 	    	 */
109 	    	disabled : null,
110 	    	/**
111 	    	 * 表单域的初始可编辑状态，如果为true则表示输入框不可用，如果存在图标，图标可以点击；否则可以正常使用(仅标签使用)
112 	    	 * @name FTextField#<ins>readonly</ins>
113 	    	 * @type Boolean
114 	    	 * @default false
115 	    	 * @example
116 	    	 * 无
117 	    	 */
118 	    	readonly : null,
119 	    	hasHiddenField : false,
120 	    	// 显示的input
121 	    	_objInput : null,
122 	    	// 隐藏的input
123 	    	_objHidden : null,
124 	    	// wrapper
125 	    	_objWrapper : null,
126 	    	_objBox : null
127 	    },
128 
129 	    _create : function() {
130 	    	this.options.id = this.element.attr("id");;
131 	    },
132 
133 	    _init : function() {
134 	    	var self = this;
135 	    },
136 	    
137 	    getBoxEl : function() {
138 	    	var op = this.options;
139 	    	op._objBox = op._objBox || this.getWrapperEl().parent();
140 	    	return op._objBox;
141 	    },
142 	    
143 	    getInputEl : function() {
144 	    	var op = this.options;
145 	    	if (op._objInput) {
146 	    		return op._objInput;
147 	    	} else {
148 	    		if (op.hasHiddenField) {
149 	    			op._objInput = this.element.next();
150 	    		} else {
151 	    			op._objInput = this.element;
152 	    		}
153 	    		return op._objInput;
154 	    	}
155 	    },
156 	    
157 	    getHiddenEl : function() {
158 	    	var op = this.options;
159 	    	if (op._objHidden) {
160 	    		return op._objHidden;
161 	    	} else {
162 	    		if (op.hasHiddenField) {
163 	    			op._objHidden = this.element;
164 	    		} else {
165 	    			op._objHidden = $I("___NOT_EXSITED___");
166 	    		}
167 	    		return op._objHidden;
168 	    	}
169 	    },
170 	    
171 	    getWrapperEl : function() {
172 	    	var op = this.options;
173 	    	op._objWrapper = op._objWrapper || this.element.parent();
174 	    	return op._objWrapper;
175 	    },
176 	    
177 	    hasHiddenField : function() {
178 	    	return (this.getHiddenEl().size() != 0);
179 	    },
180 
181 	    /**
182 	     * 设置/获取显示域的值，如果传入了参数则表示设置；该方法会始终返回当前显示域的值。<br/>
183 	     * <span style="color:red">注意：</span>不建议绑定输入框的 change或者onChange事件，因为这些事件
184 	     * 在通过js修改输入框的value之后<b>不会触发</b>（很奇妙的是，在IE6/7/8上居然触发了），所以在这种情况下请绑定 onValueChanged事件。<br/>
185 	     * 另外在默认情况下，调用此方法对输入框赋值，会触发onValueChanged事件，可以通过传入force=true来关闭事件的触发。
186 	     * @name FTextField#displayValue
187 	     * @function
188 	     * @param v 要设置的显示域的值
189 	     * @param force 非必须，如果传入true则不会触发onValueChanged事件
190 	     * @return 当前显示域的值
191 	     * @example
192 	     */
193 	    displayValue : function(v, force) {
194 	    	var inputEl = this.getInputEl();
195 	    	if (v || v=="") {
196 	    		inputEl.val(v);
197 	    		if (force !== true) { // 只有在force为true的时候才不会触发事件
198 		    		inputEl.trigger("onValueChanged");
199 	    		}
200 	    	}
201 	    	return inputEl.val();
202 	    },
203 
204 	    // 不建议使用，推荐使用setValue或者getValue
205 	    value : function(v, force) {
206 	    	var self = this;
207 	    	var op = this.options;
208 	    	if (self.hasHiddenField()) {
209 		    	var hiddenEl = this.getHiddenEl();
210 		    	if (v || v=="") {
211 		    		hiddenEl.val(v);
212 		    		if (force !== true) { // 只有在force为true的时候才不会触发事件
213 		    			hiddenEl.trigger("onValueChanged", v);
214 		    		}
215 		    	}
216 		    	return hiddenEl.val();
217 	    	} else {
218 	    		return self.displayValue(v);
219 	    	}
220 	    },
221 
222 	    /**
223 	     * 设置表单输入框的值，如果传入了参数则表示设置；该方法会始终返回当前表单输入框的值；如果存在隐藏域则访问隐藏域，否则效果同方法displayValue(v)。<br/>
224 	     * <span style="color:red">注意：</span>不建议绑定输入框的 change或者onChange事件，因为这些事件
225 	     * 在通过js修改输入框的value之后<b>不会触发</b>（很奇妙的是，在IE6/7/8上居然触发了），所以在这种情况下请绑定 onValueChanged事件。<br/>
226 	     * 另外在默认情况下，调用此方法对输入框赋值，会触发onValueChanged事件，可以通过传入force=true来关闭事件的触发。
227 	     * @name FTextField#setValue
228 	     * @function
229 	     * @param v 要设置的表单输入框的值
230 	     * @param force 非必须，如果传入true则不会触发onValueChanged事件
231 	     * @return 当前表单输入框的值，如果进行了设置，那么返回设置之后表单输入框的值
232 	     * @example
233 	     */
234 	    setValue : function(v, force) {
235 	    	return this.value(v, force);
236 	    },
237 
238 	    /**
239 	     * 获取表单域的值
240 	     * @name FTextField#getValue
241 	     * @function
242 	     * @return 当前表单域的值
243 	     * @example
244 	     */
245 	    getValue: function() {
246 	    	return this.value();
247 	    },
248 
249 	    /**
250 	     * 获取/设置表单输入框的name属性，该属性对应到表单提交时，表单域的key值
251 	     * @name FTextField#name
252 	     * @function
253 	     * @param n 要设置的表单域的名字
254 	     * @return 当前表单域的name属性，如果进行了设置，则返回设置后的值。
255 	     * @example
256 	     */
257 	    name : function(n) {
258 	    	var self = this;
259 	    	if (self.hasHiddenField()) {
260 		    	var hiddenEl = self.getHiddenEl();
261 		    	if (n || n=="") {
262 		    		hiddenEl.attr("name", n);
263 		    	}
264 		    	return hiddenEl.attr("name");
265 	    	} else {
266     			var inputEl = self.getInputEl();
267 	    		if (n || n=="") {
268 	    			inputEl.attr("name", n);
269 	    		}
270 	    		return inputEl.attr("name");
271 	    	}
272 	    },
273 
274 	    /**
275 	     * 重置表单域的值为空，如果存在隐藏域，隐藏域也会被清空。但是不会去除表单域的校验信息，如果需要去除表单域的校验信息，请调用FForm的reset方法。
276 	     * @name FTextField#reset
277 	     * @function
278 	     * @return void
279 	     * @example
280 	     */
281 	    reset : function() {
282 	    	var self = this;
283 	    	if (self.hasHiddenField()) {
284 	    		self.displayValue("");
285 	    		self.value("");
286 	    	} else {
287 	    		self.displayValue("");
288 	    	}
289 	    },
290 	    
291 	    // 不推荐使用，推荐使用setDisabled或者isDisabled方法
292 	    disabled : function(state) {
293 	    	var op = this.options;
294 	    	if (op.disabled == null) {
295 	    		op.disabled = this.getBoxEl().hasClass("f-state-disabled");
296 	    	}
297 
298 	    	var hasHiddenField = this.hasHiddenField();
299 	    	var hiddenEl = this.getHiddenEl();
300 	    	var inputEl = this.getInputEl();
301 	    	var boxEl = this.getBoxEl();
302 	    	if (arguments.length != 0) {
303 		    	if (state === true) {
304 		    		boxEl.removeClass("f-state-focus f-state-hover").addClass("f-state-disabled");
305 		    		inputEl.attr("disabled", true);
306 		    		if (hasHiddenField) {
307 		    			hiddenEl.attr("disabled", true);
308 		    		}
309 		    	} else {
310 		    		boxEl.removeClass("f-state-focus f-state-hover f-state-disabled");
311 		    		inputEl.attr("disabled", false);
312 		    		if (hasHiddenField) {
313 		    			hiddenEl.attr("disabled", true);
314 		    		}
315 		    	}
316 		    	op.disabled = (inputEl.attr("disabled") == "disabled");
317 	    	}
318 	    	return op.disabled;
319 	    },

321 	    /**
322 	     * 设置输入框的可用状态；表单处于不可用状态时，不可编辑，该表单域也不会参与表单的提交
323 	     * @name FTextField#setDisabled
324 	     * @function
325 	     * @param state 如果传入true，则表示将组件置为不可用状态；如果传入false，则表示置为可用状态。
326 	     * @return 当前表单的可用状态，true表示表单不可用，false表示表单可用
327 	     * @example
328 	     */
329 	    setDisabled : function(state) {
330 	    	return this.disabled(state);
331 	    },
332 
333 	    /**
334 	     * 获取输入框的可用状态；表单处于不可用状态时，不可编辑，该表单域也不会参与表单的提交
335 	     * @name FTextField#isDisabled
336 	     * @function
337 	     * @return 当前表单的可用状态，true表示表单不可用，false表示表单可用
338 	     * @example
339 	     */
340 	    isDisabled : function() {
341 	    	return this.disabled();
342 	    },

344 	    readonly : function(state) {
345 	    	var op = this.options;
346 	    	if (op.readonly == null) {
347 	    		op.readonly = this.getBoxEl().hasClass("f-state-readonly");
348 	    	}
349 
350 	    	var hasHiddenField = this.hasHiddenField();
351 	    	var hiddenEl = this.getHiddenEl();
352 	    	var inputEl = this.getInputEl();
353 	    	var boxEl = this.getBoxEl();
354 	    	if (arguments.length != 0) {
355 		    	if (state === true) {
356 		    		boxEl.removeClass("f-state-focus f-state-hover").addClass("f-state-readonly");
357 		    		inputEl.attr("readonly", true);
358 		    		if (hasHiddenField) {
359 		    			hiddenEl.attr("readonly", true);
360 		    		}
361 		    	} else {
362 		    		boxEl.removeClass("f-state-focus f-state-hover f-state-readonly");
363 		    		inputEl.attr("readonly", false);
364 		    		if (hasHiddenField) {
365 		    			hiddenEl.attr("readonly", true);
366 		    		}
367 		    	}
368 		    	op.readonly = (inputEl.attr("readonly") == "readonly");
369 	    	}
370 	    	return op.readonly;
371 	    },

374 	     * 设置输入框的可编辑状态；表单处于不可编辑状态时，输入框不可用，如果存在图标，图标可以点击。
375 	     * @name FTextField#setReadonly

378 	     * @return 当前表单的可编辑状态，true表示表单输入框不可编辑，false表示表单可编辑。

386 	     * 获取输入框的可编辑状态；表单处于不可编辑状态时，输入框不可用，如果存在图标，图标可以点击。
387 	     * @name FTextField#isReadonly
388 	     * @function
389 	     * @param state 如果传入true，则表示将组件置为不可编辑状态；如果传入false，则表示置为可编辑状态。
390 	     * @return 当前表单的可编辑状态，true表示表单输入框不可编辑，false表示表单可编辑。
391 	     * @example
392 	     */
393 	    isReadonly : function() {
394 	    	return this.readonly();

399 	     * @name FPopupMessage#destroy
400 	     * @function

404 		    var op = this.options;
405 		    this.getInputEl().unbind();
406 		    this.getBoxEl().unbind();
407 		    // 调用父类的销毁方法
408 		    $.Widget.prototype.destroy.call(this);
409 	    }
