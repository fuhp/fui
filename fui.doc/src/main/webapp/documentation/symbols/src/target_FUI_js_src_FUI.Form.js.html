 15 /**
 16  * @name FForm
 17  * @class

290 	    /**
291 	     * 获取Form内有name属性且非disabled状态表单域，以对象的形式返回。 <br/>
292 	     * 对同名域的处理：
293 	     * <pre>
294 	     * ① 如果是radio的话，则在同名表单域中选取被选中的radio对应的value，比如
295 	     * <input type="radio" name="sex" value="1">
296 	     * <input type="radio" name="sex" value="2">
297 	     * <input type="radio" name="sex" value="3">
298 	     * 如果value=2的radio被选中，那么最后生成的表单域为 sex:"2"，其他几个radio会被忽略
299 	     *
300 	     * ② 非radio表单域，两个及两个以上的同名表单组装为数组，比如
301 	     * <input type="text" name="favor" value="2">
302 	     * <input type="text" name="favor" value="3">
303 	     * <input type="checkbox" name="favor" value="4">
304 	     * 如上所示，最后生成的表单域为 favor:["2", "3", "4"]，被表示为数组。
305 	     *
306 	     * 需要特别说明的是：原生的checkbox，如果没有被勾选，则也不会参与表单提交，
307 	     * 比如三个原生的checkbox，name分别为name1,name2,name2，值分别为1,2,3：
308 	     * a. 第一个被勾选，那么getValues返回{name1:"1"}
309 	     * b. 第一个第二个被勾选，返回 {name1:"1",name2:"2"}
310 	     * c. 都被勾选，那么返回 {name1:"1", name2:["2","3"]}
311 	     * d. 都不勾选，那么返回 {}，即都不参与表单提交。
312 	     * </pre>
313 	     * @name FForm#getValues
314 	     * @function
315 	     * @example
316 	     * $I('form').FForm("getValues");
317 	     */
318 	    getValues : function() {
319 	    	var self = this;
320 	    	var values = {};
321 	    	$("[name]:input", this.element).each(function() {
322 	    		var el = $(this);
323 	    		if (el.attr("disabled")) { // NO! [name]:enabled
324 	    			return;
325 	    		}
326 	    		var name = el.attr("name");
327 	    		var value = self._getValue(el);
328 	    		if (name && (value || value=="")) {
329 		    		// 处理单选框
330 		    		if (el.is(":radio,:checkbox")) {
331 		    			if (!el.is(":checked")) {
332 		    				return; // 没有被选中则忽略，浏览器会保证同名的radio只会有一个被选中
333 		    			}
334 		    		}
335 	    			var preValue = values[name];
336 	    			if (preValue || preValue =="") { // 如果传入的值非空
337 	    				if ($.isArray(preValue)) { // 如果已经是数组了则直接在后面追加
338 	    					preValue.push(value);
339 	    				} else { // 否则构造一个新的数组
340 	    					var array = [preValue, value];
341 	    					values[name] = array;
342 	    				}
343 	    			} else {
344 	    				values[name] = value;
345 	    			}
346 	    		}
347 	    	});
348 	    	return values;
349 	    },

351 	    /**
352 	     * 设置Form内有name属性且非disabled状态表单域，
353 	     * 假设传入的对象为 {key1:"value1", key2:"value2"}，那么此操作会设置name为key1和key2的表单域分别为value1和value2，
354 	     * 其他的表单域不会被修改。如果需要清空所有的表单域，请先调用 reset 方法来重置所有的表单域。
355 	     * <b>对同名域的处理</b>：
356 	     * <pre>
357 	     * ① 如果是radio的话，则在同名表单域中选取被选中的radio对应的value，比如
358 	     * <input type="radio" name="sex" value="1">
359 	     * <input type="radio" name="sex" value="2">
360 	     * <input type="radio" name="sex" value="3">
361 	     * 如果value=2的radio被选中，那么最后生成的表单域为 sex:"2"，其他几个radio会被忽略
362 	     *
363 	     * ② 非radio表单域，两个及两个以上的同名表单组装为数组，比如
364 	     * <input type="text" name="favor" value="2">
365 	     * <input type="text" name="favor" value="3">
366 	     * <input type="checkbox" name="favor" value="4">
367 	     * 如上所示，最后生成的表单域为 favor:["2", "3", "4"]，被表示为数组。
368 	     * </pre>
369 	     * @name FForm#setValues
370 	     * @param values 表单域值对象
371 	     * @function
372 	     * @example
373 	     * $I('form').FForm("setValues", {key:"value1", key2:"value2"});
374 	     */
375 	    setValues : function(values) {
376 	    	var self = this;
377 	    	$("[name]:input", this.element).each(function() {
378 	    		var el = $(this);
379 	    		if (el.attr("disabled")) { // NO! [name]:enabled
380 	    			return;
381 	    		}
382 	    		var name = el.attr("name");
383 	    		var value = values[name];
384 	    		if ($.isArray(value)) {
385 	    			if (value.length > 0) {
386 	    				value = value.shift();
387 	    			} else {
388 	    				values[name] = undefined;
389 	    				value = "";
390 	    			}
391 	    		}
392 	    		if (value || value=="") {
393 	    			if (el.is(":radio,:checkbox")) {
394 	    	    		var myValue = self._getValue(el);
395 	    	    		if (myValue == value) {
396 	    	    			el.attr("checked", true);
397 	    	    		} else {
398 	    	    			el.attr("checked", false);
399 	    	    		}
400 	    	    		return;
401 	    			}
402 	    			self._setValue(el, value);
403 	    		}
404 	    	});
405 	    },

483 	    /**
484 	     * form提交
485 	     * @name FForm#submit
486 	     * @param params 附加的数据，以json对象表示
487 	     * @param options ajax请求的选项，除了success、error、beforeSend等属性外，都延用了FUI的FAjax组件，更详细的内容请见
488 	     * <a href='./FAjax.html' target='_blank'>FAjax组件</a>
489 	     * @function
490 	     * @example
491 	     * $F('form').FForm("submit");
492 	     */
493 	    submit : function(params, options) {
494 	    	var self = this;
495 	    	var op = this.options;
496 	    	var iAsyn = true;
497 	    	var specialParams = params || op.params;
498 	    	var data = op.params || {};
499 	    	data = $.extend({}, self.getValues(), (specialParams || {}));
500 	    	var action = op.action || self.element.attr("action");
501 	    	var ajaxOpts = $.extend({},
502     			{
503     	            url : action,	// 要访问的url
504     	            data : data,		// 请求参数
505     				success : function(data, textStatus, jqXHR) {
506     					var func = op.onSuccess;
507     					if (func && !$.isFunction(func)) {
508     						func = self._parseFunc(func);
509     						if (func) {
510         						op.onSuccess = func;
511     						}
512     					}
513     					if (func) {
514     						return func.call(self, data, textStatus, jqXHR);
515     					}
516     				},
517     				failure : function(data, textStatus, jqXHR) {
518     					var func = op.onFailure;
519     					if (!$.isFunction(func)) {
