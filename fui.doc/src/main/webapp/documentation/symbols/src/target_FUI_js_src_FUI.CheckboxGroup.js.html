 24 /**

 33 /**

 36  * @type String
 37  * @default ""
 38  * @example

 42 /**
 43  * 表单域的初始可用状态，如果为true则表示不可用，也不会参与表单提交；否则可以正常使用(仅标签使用)

 45  * @type Boolean
 46  * @default false

 51 /**
 52  * 表单域的初始可编辑状态，如果为true则表示输入框不可用，如果存在图标，图标可以点击；否则可以正常使用(仅标签使用)
 53  * @name FCheckboxGroup#<ins>readonly</ins>

 61 (function($) {
 62 	// FCheckboxGroup对应的DOM必须是一个Checkbox的容器
 63 	// 有且仅有一个隐藏input具有name属性，其他的checkbox都必须有value属性，如果没有则会被忽略
 64 	$.widget("FUI.FCheckboxGroup", {
 65 	    options : {
 66             /**
 67              * seperator用来分隔被选中的checkbox的value值，作为表单提交的值。
 68              * @name FCheckboxGroup#<ins>seperator</ins>
 69              * @type String
 70              * @default ","
 71              * @example
 72              * <pre>
 73              * $("#checkboxGroup1").FCheckboxGroup({seperator:":"});
 74              * 此时，如果value为"0"、"1"、"3"的checkbox被选中，那么该组件的进行表单提交的值为 "0:1:3"
 75              * </pre>
 76              */
 77 	    	seperator: ",",
 78             /**
 79              * 组件的默认值，在组件初始化或者调用reset的时候，会采用此默认值初始化checkbox的选中状态，
 80              * 比如defaultValue="0,1,3"，那么value为0,1,3的checkbox会被选中。注意：defaultValue的值分隔串必须与seperator属性一致。
 81              * @name FCheckboxGroup#<ins>defaultValue</ins>
 82              * @type String
 83              * @default ""
 84              * @example
 85              */
 86 	    	defaultValue: "",
 87 
 88 	    	_objBox : null
 89 	    },
 90 
 91 	    _create : function() {
 92 		    // 初始化ID
 93 		    this.options.id = this.element.attr("id");
 94 		    this.options._objBox = null;
 95 	    },
 96 
 97 	    _init : function() {
 98 		    this.options._objBox = null;
 99 		    // 获取所有的checkbox组件绑定click事件
100 	    	this._bindEvent();
101 	    	// 如果设置了默认值，则使用默认值初始化
102 	    	if (this.options.defaultValue) {
103 	    		this.reset(); // 使用defaultValue初始化组件
104 	    	}
105 	    	// add 20130123 hanyin 增加初始状态
106 	    	if (this.options.readonly === true) {
107 	    		this.readonly(true);
108 	    	}
109 	    	if (this.options.disabled === true) {
110 	    		this.disabled(true);
111 	    	}
112 	    	// end add 20130123 hanyin 增加初始状态
113 	    },
114 	    
115 	    _bindEvent : function() {
116 	    	var self = this;
117 	    	$(":checkbox[value]", this._getBoxEl()).click(function() {
118 	    		if (self.readonly() || self.disabled()) { // 避免在readonly状态下，checkbox依然可选
119 		    		return false;
120 	    		}
121     			self._triggerValueChange();
122 	    	});
123 	    },
124 	    
125 	    _getBoxEl : function() {
126 	    	var selfEl = this.element;
127 	    	var op = this.options;
128 	    	if (op._objBox == null) {
129 	    		if (selfEl.is("input")) {
130 	    			op._objBox = selfEl.parent();
131 	    		} else { // div等容器
132 	    			op._objBox = selfEl;
133 	    		}
134 	    	}
135 	    	return op._objBox;
136 	    },
137 	    
138 	    _triggerValueChange : function() {
139 	    	var values = [];
140 	    	$(":checkbox[value]:checked", this._getBoxEl()).each(function() {
141 	    		var itemEl = $(this);
142 	    		if (!itemEl.attr("disabled")) { // 只有可用状态的才能被求值
143 	    			values.push($(this).val());
144 	    		}
145 	    	});
146 	    	var sep = this.options.seperator || ",";
147 	    	$("input[name]", this._getBoxEl()).val(values.join(sep));
148 	    },
149 
150 	    value : function(v, force) {
151 	    	if (v || v=="") {
152 		    	var sep = this.options.seperator || ",";
153 	    		var array = v;
154 	    		if ($Utils.isString(v)) {
155 	    			array = v.split(sep);
156 	    		}
157 	    		if ($Utils.isArray(array)) {
158 	    			this._setValueByArray(array);
159 	    			if (force !== true) {
160 	    				this.element.trigger("onValueChanged");
161 	    			}
162 	    		}
163 	    	}
164 	    	return $("input[name]", this._getBoxEl()).val();
165 	    },
166 
167 	    /**
168 	     * 批量设置内部checkbox的值
169 	     * @name FCheckboxGroup#setValue
170 	     * @function
171 	     * @param v 使用seperator分隔的字符串
172 	     * @return String
173 	     * @example
174 	     * 无
175 	     */
176 	    setValue : function(v, force) {
177 	    	return this.value(v, force);
178 	    },
179 
180 	    /**
181 	     * 获取表单域的值，搜集checkbox的value值，并使用seperator属性的值分隔。
182 	     * @name FCheckboxGroup#getValue
183 	     * @function
184 	     * @return 当前表单域的值
185 	     * @example
186 	     */
187 	    getValue: function() {
188 	    	return this.value(v);
189 	    },
190 
191 	    disabled : function(state) {
192 	    	var boxEl = this._getBoxEl();
193 	    	if (arguments.length != 0) {
194 	    		state = (state===true);
195 	    		if (state) {
196 	    			boxEl.addClass("f-state-disabled");
197 	    		} else {
198 	    			boxEl.removeClass("f-state-disabled");
199 	    		}
200 	    		$(":checkbox[value],input[name]", boxEl).attr("disabled", state);
201 	    	}
202 	    	return boxEl.hasClass("f-state-disabled");
203 	    },
204 
205 	    /**
206 	     * 设置本组件的禁用状态，true表示禁用不能参与表单提交
207 	     * @name FCheckboxGroup#setDisabled
208 	     * @function
209 	     * @param state 传入true表示修改组件为禁用状态；传入false则恢复组件的状态；不传则不会修改状态
210 	     * @return Boolean
211 	     * @example
212 	     * 无
213 	     */
214 	    setDisabled : function(state) {
215 	    	return this.disabled(state);
216 	    },
217 
218 	    /**
219 	     * 获取本组件的禁用状态，true表示禁用不能参与表单提交
220 	     * @name FCheckboxGroup#isDisabled
221 	     * @function
222 	     * @param state 传入true表示修改组件为禁用状态；传入false则恢复组件的状态；不传则不会修改状态
223 	     * @return Boolean
224 	     * @example
225 	     * 无
226 	     */
227 	    isDisabled : function() {
228 	    	return this.disabled();
229 	    },
230 
231 	    readonly : function(state) {
232 	    	var boxEl = this._getBoxEl();
233 	    	if (arguments.length != 0) {
234 	    		state = (state===true);
235 	    		if (state) {
236 	    			boxEl.addClass("f-state-readonly");
237 	    		} else {
238 	    			boxEl.removeClass("f-state-readonly");
239 	    		}
240 	    		$(":checkbox[value],input[name]", this.boxEl).attr("readonly", state);
241 	    	}
242 	    	return boxEl.hasClass("f-state-readonly");
243 	    },
244 
245 	    /**
246 	     * 设置本组件的只读状态，true表示组件不可通过鼠标点击来修改checkbox的选中状态
247 	     * @name FCheckboxGroup#readonly
248 	     * @function
249 	     * @param state 传入true表示修改组件为只读状态；传入false则恢复组件的状态；不传则不修改组件状态
250 	     * @return Boolean
251 	     * @example
252 	     * 无
253 	     */
254 	    setReadonly : function(state) {
255 	    	return this.readonly(state);
256 	    },
257 
258 	    /**
259 	     * 获取本组件的只读状态，true表示组件不可通过鼠标点击来修改checkbox的选中状态
260 	     * @name FCheckboxGroup#readonly
261 	     * @function
262 	     * @param state 传入true表示修改组件为只读状态；传入false则恢复组件的状态；不传则不修改组件状态
263 	     * @return Boolean
264 	     * @example
265 	     * 无
266 	     */
267 	    isReadonly : function() {
268 	    	return this.readonly();
269 	    },
270 	    
271 	    /**
272 	     * 使用defaultValue来重置组件，如果没有设置defaultValue属性，则将所有的checkbox都置为非选中状态
273 	     * @name FCheckboxGroup#reset
274 	     * @function
275 	     * @example
276 	     * 无
277 	     */
278 	    reset : function() {
279 	    	var defaultValue = this.options.defaultValue || "";
280 	    	this.value(defaultValue);
281 	    },
282 
283 	    /**
284 	     * 获取本表单的名字，对应表单域的key值
285 	     * @name FCheckboxGroup#name
286 	     * @function
287 	     * @return String
288 	     * @example
289 	     * 无
290 	     */
291 	    name : function(n) {
292 	    	var hidden = $("input[name]", this._getBoxEl());
293 	    	if (n || n=="") {
294 	    		hidden.attr("name", n);
295 	    	}
296 	    	return hidden.attr("name");
297 	    },
298 	    
299 	    _setValueByArray : function(array) {
300 	    	$(":checkbox[value]", this._getBoxEl()).each(function() {
301 	    		var self = $(this);
302 	    		var value = self.val();
303 	    		if ($.inArray(self.val(), array) >= 0) {
304 	    			self.attr("checked", true);
305 	    		} else {
306 	    			self.attr("checked", false);
307 	    		}
308 	    	});
309 			this._triggerValueChange();
310 	    },
311 
312 	    /**
313 	     * 销毁组件
314 	     * @name FCheckboxGroup#destroy
315 	     * @function
316 	     * @return Object
317 	     * @example
318 	     * 无
319 	     */
320 	    destroy : function() {
321 	    	$(":checkbox[value]", this._getBoxEl()).unbind();
322 	    	this.options._objBox = null;
323 	    }
324 	});
325 })(jQuery);
