 82 	$.FUI.FDate = function (){
 83 		// 年 月 日 时 分 秒
 84 		this.y=0, this.M=0,this.d=0,this.H= 0,this.m = 0,this.s = 0;
 85 		this.lastFmt = "";
 86 		this.lastDateStr = "";
 87 		
 88 		this._reset = function() {
 89 			this.y = this.M = this.d = this.H = this.m = this.s = 0;
 90 		};
 91 		
 92 		// 将传入的字符串按照fmt格式转换
 93 		this.parseStr =function(str, fmt) {
 94 			var self = this;
 95 			var year, month, day, hour, minite, second;
 96 			fmt = fmt || "yyyy-MM-dd";
 97 
 98 			self.lastFmt = fmt;
 99 			self.lastDateStr = str;
100 			
101 			for (var i=0; i<fmt.length;) {
102 				var literal = self._parseLiteral(fmt, i);
103 				if (literal.literal == null) {
104 					break;
105 				}
106 				var index = literal.startIndex;
107 				var count = literal.count;
108 				// 不能用parseInt，他无法正确转换 "09"、"01"等0开头的字符串
109 				self[literal.literal] = Number(str.substr(index, count), 10);
110 				i = index + count;
111 			}
112 		};
113 		
114 		this._parseLiteral= function(fmt, start) {
115 			var literal = null;
116 			var count = 0;
117 			var startIndex = -1;
118 			var value = "";
119 			for (var i=start; i<fmt.length; i++) {
120 				var ch = fmt.charAt(i);
121 				if (literal != null) {
122 					if (literal == ch) {
123 						count ++;
124 						value = value + ch;
125 					} else {
126 						break;
127 					}
128 				} else {
129 					if (ch=="y" || ch=="M" || ch=="d" || ch=="H" || ch=="m" || ch=="s") {
130 						literal = ch;
131 						count++;
132 						value = value + ch;
133 						startIndex = i;
134 					}
135 				}
136 			}
137 			return {literal:literal, count:count, startIndex: startIndex, value: value};
138 		};
139 
140 		// 将日期组件按照指定的格式转换为字符串
141 		this.toString = function(fmt) {
142 			var self = this;
143 			fmt = fmt || "yyyy-MM-dd";
144 			if (self.lastFmt == fmt) {
145 				return self.lastDateStr;
146 			}
147 			var result = fmt;
148 			for (var i=0; i<fmt.length;) {
149 				var lit = self._parseLiteral(fmt, i);
150 				if (lit.literal == null) {
151 					break;
152 				}
153 				var literal = lit.literal;
154 				var count = lit.count;
155 				var fmtValue = lit.value;
156 				var value = self._getLiteralValue(literal, count);
157 				result = result.replace(fmtValue, value);
158 
159 				i = lit.startIndex + count;
160 			}
161 			return result;
162 		};
163 
164 		this._getLiteralValue = function(literal, length) {
165 			var self = this;
166 			var value = "" + self[literal];
167 			var gap = length - value.length;
168 			if (gap > 0) {
169 				for (var i=0; i<gap; i++) {
170 					value = "0" + value;
171 				}
172 			}
173 			return value;
174 		};
175 	};

243 		    		$I(op.id + "-input").val("");
244 	    		} else {
245 		    		var date = new $.FUI.FDate();
246 		    		var realDateFmt = op.realDateFmt || (op.showOpts||{}).realDateFmt;
247 		    		var realTimeFmt = op.realTimeFmt || (op.showOpts||{}).realTimeFmt;
248 		    		var realFullFmt = op.realFullFmt || (op.showOpts||{}).realFullFmt;
249 		    		var dateFmt = op.dateFmt || (op.showOpts||{}).dateFmt;
250 		    		if (realDateFmt) {
251 		    			fmt = realDateFmt;
252 		    		} else if (realTimeFmt) {
253 		    			fmt = realTimeFmt;
254 		    		} else {
255 		    			fmt = realFullFmt;
256 		    		}
257 		    		date.parseStr(v, fmt);
258 		    		$I(op.id + "-input").val(date.toString(dateFmt));
259 	    		}
