 65 	    show : function(op, callback, force) {
 66 	    	if (!this.options._isValid) {
 67 	    		return;
 68 	    	}
 69 	    	op = op || {};
 70 	    	// force true表示忽略组件的默认值，否则会将传入的op与默认值合并
 71 	    	var option = this._uniformOption(op, force);
 72 	    	this.options._curOptions = option;
 73 	    	//{ 20121129 hanyin 将回调放到showOpts中
 74 	    	option.show = $.extend({}, (op.show || {}));
 75 	    	option.show.callback = callback || (op.show || {}).callback;
 76 	    	//} 
 77 
 78 	    	var showMode = this._parseShowMode(option);
 79 	    	var result = false;
 80 	    	if (showMode === "ATTACH") {
 81 	    		var attachEl = option.attach;
 82 	    		if (!$Utils.isJQueryObj(option.attach)) { // 如果传入的不是jquery对象，则认为是选择器
 83 		    		attachEl = $(option.attach);
 84 	    		}
 85 	    		if (attachEl.size() == 0) {
 86 	    			return false;
 87 	    		}
 88 		    	// 依附组件相对于document的位置
 89 		    	var attachOffset = attachEl.offset();
 90 		    	// 依附组件的大小
 91 		    	var attachSize = this._calAttachSize(attachEl);
 92 		    	// 外框容器的大小，用于检测碰撞
 93 		    	var containerBox = this._calContainerBox(option.container);
 94 	    		var realOffset = this._calAttachMode(attachOffset, attachSize,
 95 	    				option.size, option.direction, option.offset, containerBox, option.collision);
 96 	    		result = this._showAbsolute(realOffset);
 97 	    	} else if (showMode === "ABSOLUTE") {
 98 		    	// 依附组件相对于document的位置
 99 		    	var attachOffset = {left:option.position.left, top:option.position.top};
100 		    	// 依附组件的大小
101 		    	var attachSize = {height:0,width:0};
102 		    	// 外框容器的大小，用于检测碰撞
103 		    	var containerBox = this._calContainerBox(option.container);
104 		    	if (attachOffset.left == "center") {
105 		    		// x-min = max-(x+lenght) -> x=(max+min-lenght)/2
106 		    		attachOffset.left = (containerBox.minX + containerBox.maxX - option.size.width)/2;
107 		    	}
108 		    	if (attachOffset.top == "center") {
109 		    		// x-min = max-(x+lenght) -> x=(max+min-lenght)/2
110 		    		attachOffset.top = (containerBox.minY + containerBox.maxY - option.size.height)/2;
111 		    	}
112 	    		var realOffset = this._calAttachMode(attachOffset, attachSize, 
113 	    				option.size, option.direction, option.offset, containerBox, option.collision);
114 	    		result = this._showAbsolute(realOffset);
115 	    	} else if (showMode === "FIXED") {
116 	    		result = this._calFixedMode(option.size, option.offset);
117 	    	} else { // 显示在屏幕正中间
118 	    		result = this._calOtherMode(option);
119 	    	}
120 	    	return result;
121 	    },

123 	    // 依附于某一个组件
124 	    // attachOffset: 依附组件相对于document的偏移量
125 	    // attachSize: 依附组件的大小
126 	    // size: 弹出组件的大小
127 	    // direction: 弹出组件的弹出方向 left、right、up、down
128 	    // offset: 弹出组件相对于依附组件的偏移量
129 	    // containerSize: 外框容器的大小
130 	    _calAttachMode : function(attachOffset, attachSize, size, direction, offset, containerBox, collision) {
131 	    	offset = offset || {left:0,top:0};
132 	    	direction = direction || ['down', 'up'];
133 	    	if (direction.length == 0) {
134 	    		direction = ['down', 'up', "right", "upright", "left", "upleft"];
135 	    	}
136 
137 	    	// 遍历所有的方向
138 	    	var resultDirection = null;
139 	    	var firstValidDirection = null
140 	    	var resultPos = null;
141 	    	for (var i=0; i<direction.length; i++) {
142 	    		var dir = direction[i].toLowerCase();
143 	    		var tryFunc = this["_try" + dir];
144 	    		if ($.isFunction(tryFunc)) {
145 	    			if (firstValidDirection == null) {
146 	    				firstValidDirection = dir;
147 	    			}
148 	    			var result = tryFunc.call(this, attachOffset, attachSize, size, containerBox);
149 	    			if (result === false) {
150 	    				continue;
151 	    			} else {
152 	    				resultDirection = dir;
153 	    				resultPos = result;
154 	    				break;
155 	    			}
156 	    		}
157 	    	}
158 	    	// 如果所有的方向都不能满足碰撞检测，则选取第一个有效的方向，如果没有合适的方向，则抛出异常
159 	    	if (resultDirection == null) {
160 	    		if (firstValidDirection != null) {
161 	    			resultDirection = firstValidDirection;
162 	    			resultPos = this["_cal" + firstValidDirection](attachOffset, attachSize, size, containerBox, collision);
163 	    	    	if (collision == "fit") {
164 	    	    		result = this._adjustPos(resultPos, size, containerBox);
165 	    	    	}
166 	    		} else {
167 	    			throw new Error("FPopupBox: no valid direction");
168 	    		}
169 	    	}
170 	    	var result = null;
171 	    	if (resultPos != null) {
172 			    var resultOffset = this._parseRealOffset(offset, resultDirection);
173 			    result = this._calRealOffset(resultPos, resultOffset);
174 	    	}
175 	    	return result;
176 	    },

178 	    _adjustPos : function(pos, size, containerBox) {
179 	    	var minX = pos.left;
180 	    	var minY = pos.top;
181 	    	var maxX = minX + size.width;
182 	    	var maxY = minY + size.height;
183 	    	if (minX < containerBox.minX) {
184 	    		pos.left = containerBox.minX;
185 	    	} else if (maxX > containerBox.maxX) {
186 	    		pos.left = containerBox.maxX - size.width;
187 	    	}
188 	    	if (minY < containerBox.minY) {
189 	    		pos.top = containerBox.minY;
190 	    	} else if (maxY > containerBox.maxY) {
191 	    		pos.top = containerBox.maxY - size.height;
192 	    	}
193 	    	return pos;
194 	    },

200 	    	return realOffset;

210 	    // 计算容器相对于document的最大最小位置
211 	    _calContainerBox : function(el) {
212 	    	var docEl = $(document);
213 	    	var winEl = $(window);
214 	    	if (el == null || !$Utils.isJQueryObj(el) || el.size() == 0) {
215 	    		el = docEl;
216 	    	}
217 	    	var box = {};
218 	    	if (el.get(0).window == window) {
219 		    	var scrollX = docEl.scrollLeft();
220 		    	var scrollY = docEl.scrollTop();
221 		    	box.minX = scrollX;
222 		    	box.minY = scrollY;
223 		    	box.maxX = scrollX + winEl.outerWidth();
224 		    	box.maxY = scrollY + winEl.outerHeight();
225 	    	} else if (el.get(0) == document) {
226 	    		var docWidth = docEl.width();
227 	    		var docHeight = docEl.height();
228 	    		var winWidth = winEl.width();
229 	    		var winHeight = winEl.height();
230 		    	box.minX = 0;
231 		    	box.minY = 0;
232 		    	// 适用于document比window小的情况，此时以window为准；出现滚动条则以document为准
233 		    	box.maxX = docWidth<winWidth?winWidth:docWidth;  
234 		    	box.maxY = docHeight<winHeight?winHeight:docHeight;
235 	    	} else {
236 		    	var elWidth = el.width();
237 		    	var elHeight = el.height();
238 		    	var offset = el.offset() || {left:0, top:0};
239 		    	box.minX = offset.left;
240 		    	box.minY = offset.top;
241 		    	box.maxX = offset.left + el.outerWidth();
242 		    	box.maxY = offset.top + el.outerHeight();
243 	    	}
244 	    	return box;
245 	    },

278 	    	var maxY = attachOffset.top + attachSize.height + size.height;
279 	    	if (maxY > containerBox.maxY) {

281 	    	}

283 	    	if (maxX > containerBox.maxX) {
284 	    		return false;
285 	    	}
286 	    	return this._caldown(attachOffset, attachSize, size);
287 	    },

389 	    _showAt : function(offset, position) {
390 		    var op = this.options;
391 		    position = position || "absolute";
392 		    op._objPopup.css("position", position);
393 		    if (offset.left != null && !isNaN(offset.left)) {
394 			    op._objPopup.css("left", offset.left + "px");
395 		    } else {
396 			    op._objPopup.css("left", "");
397 		    }
398 		    if (offset.right != null && !isNaN(offset.right)) {
399 		    	op._objPopup.css("right", offset.right + "px");
400 		    } else {
401 		    	op._objPopup.css("right", "");
402 		    }
403 		    if (offset.top != null && !isNaN(offset.top)) {
404 		    	op._objPopup.css("top", offset.top + "px");
405 		    } else {
406 		    	op._objPopup.css("top", "");
407 		    }
408 		    if (offset.bottom != null && !isNaN(offset.bottom)) {
409 		    	op._objPopup.css("bottom", offset.bottom + "px");
410 		    } else {
411 		    	op._objPopup.css("bottom", "");
412 		    }
413 		    var zIndex = op.zIndex || this._getZIndex();
414 		    op._objPopup.css("z-index", zIndex);
415 		    // op._objPopup.show(); // jquery会把 ul等元素解释为inline
416 		    var showOp = this.options._curOptions.show;
417 		    var callback = null;
418 		    if (showOp != null && typeof showOp === 'object') {
419 		    	callback = showOp["callback"];
420 		    	var effect = showOp["effect"];
421 		    	var styles = showOp["styles"];
422 		    	if (effect != null) {
423 		    		if (this._doEffectAnimate(op._objPopup, effect, showOp) === true) {
424 		    			return true;
425 		    		}
426 		    	} else if (styles != null) {
427 		    		if (this._doStylesAnimate(op._objPopup, showOp) === true) {
428 		    			return true;
429 		    		}
430 		    	}
431 		    }
432 	    	op._objPopup.css("display", "block");
433 	    	if ($.isFunction(callback)) {
434 	    		try {
435 	    			callback.call(this, op);
436 	    		} catch (e) {
437 	    			// 忽略
438 	    			alert(e);
439 	    		}
440 	    	}
441 	    	return true;
442 	    },

444 	    _doEffectAnimate : function(el, effect, op) {
445 	    	var func = el[effect];
446 	    	if (func != null) {
447 	    		el.stop(true); // 删除动画队列，避免误操作
448 		    	if (effect == "fadeTo") {
449 		    		el[effect](op.duration, op.opacity, op.callback);
450 		    	} else {
451 		    		el[effect](op.duration, op.callback);
452 		    	}
453 		    	return true;
454 	    	}
455 	    },

460 	    	return true;

475 	    _calOtherMode : function(op) {
476 	    	var position = op.position;
477 	    	if (position === "center") {
478 		    	// 外框容器的大小，用于检测碰撞
479 		    	var containerSize = this._calContainerSize($(window));
480 		    	var popupSize = op.size;
481 		    	var offset = {};
482 		    	offset.left = (containerSize.width -popupSize.width)/2;
483 		    	if (offset.left < 0) {
484 		    		offset.left = 0;
485 		    	}
486 		    	offset.top = (containerSize.height -popupSize.height)/2;
487 		    	if (offset.top < 0) {
488 		    		offset.top = 0;
489 		    	}
490 		    	this._showAt(offset, "fixed");
491 	    	}

494 	    _parseShowMode : function(option) {
495 	    	if (option.attach != null && $.trim(option.attach).length != 0) {
496 	    		return "ATTACH"; // 依附于某一个组件
497 	    	}
498 	    	if (option.position != null) {
499 	    		var position = option.position;
500 	    		if (typeof position=='object' && 
501 	    				(position.left != null || position.top != null 
502 	    						|| position.bottom != null || position.right != null)) {
503 	    			return "ABSOLUTE"; // 绝对布局
504 	    		}
505 	    	}
506 	    	if (option.position === "fixed") {
507 	    		return "FIXED"; // 固定布局
508 	    	}
509 	    	return option.position; //未定义处理方式
510 	    },

512 	    _uniformOption : function(op, force) {
513 	    	op = op || {};
514 	    	var option = null;
515 	    	if (force === true) {
516 	    		option = $.extend(false, {}, op); // 参数浅拷贝
517 	    	} else {
518 	    		option = $.extend(false, {}, op);
519 	    		option = $Utils.applyIf(option, this.options); // 合并默认值
520 	    	}
521 
522 	    	option.direction = this._parseDirection(option.direction);
523 	    	option.offset = this._parseOffset(option.offset);
524 	    	option.position = this._parsePosition(option.position);
525 	    	option.size = this._calPopupBoxSize(option);
526 	    	option.container = this._calContainer(option);
527 	    	return option;
528 	    },

534 	    	}

538 	    _calPopupBoxSize : function(option) {
539 	    	var popupBoxEl = this.options._objPopup;
540 	    	var size = {};
541 	    	var width = parseInt(option.width || "");
