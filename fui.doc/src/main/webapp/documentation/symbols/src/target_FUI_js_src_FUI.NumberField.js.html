 35 (function($, undefined) {
 36 	// focus事件时，将隐藏域的值拷贝到显示域中；blur事件，将显示域的值按照要求保存在隐藏域，显示域的内容为隐藏域格式化之后的值

 38     $.widget('FUI.FNumberField', {
 39         options: {
 40             /**
 41              * 设置小数点的最大位数。默认值为：2，即小数点最大位数是2。如果设置为0，则小数点也无法输入。
 42              * @name FNumberField#decimals
 43              * @type Number
 44              * @default 2
 45              * @example
 46              * 无
 47              */
 48             decimals : 2,
 49             /**
 50              * 是否以金额格式显示数字。默认值为：false，即不以金额形式显示数字，如果设置为true，则以金额格式显示数字，例如：1,111,111.00。
 51              * @name FNumberField#moneyFormat
 52              * @type Boolean
 53              * @default false
 54              * @example
 55              * 无
 56              */
 57             moneyFormat : false,
 58             /**
 59              * 设置组件的默认值
 60              * @name FNumberField#defaultValue
 61              * @type String/Number
 62              * @default ""
 63              * @example
 64              * 无
 65              */
 66             defaultValue : null,
 67             /**
 68              * 在输入值变化之后，焦点移开输入框，调用此方法
 69              * @name FNumberField#afterChange
 70              * @type String
 71              * @default null
 72              * @example
 73              * function afterChange(value, prevValue) {
 74              *     ("#console").append("[" + value + ":" + prevValue + "]<br>");
 75              * }
 76              * 
 77              * value：为当前的值；prevValue：为上次的值，如果上次为空，则为""。
 78              * 无
 79              */
 80             afterChange: null
 81         },
 82         
 83         MAX_LENGTH : 15,
 84         
 85         // 组件级别的创建，先关事件的绑定
 86         _create : function() {
 87         	var op = this.options;
 88         	op.id = this.element.attr("id");
 89             this._bindEvent();	// 绑定事件
 90         },
 91         
 92         _init : function() {
 93         	// 初始化处理
 94         	var op = this.options;
 95         	// 初始化参数
 96         	this.setDecimals(op.decimals);
 97         	var moneyFormat = op.moneyFormat;
 98         	if (moneyFormat == true || moneyFormat == "true") {
 99         		op.moneyFormat = true;
100         	} else {
101         		op.moneyFormat = false;
102         	}
103         	// 初始化默认值
104         	if (op.defaultValue) {
105         		this.reset();
106         	}
107         },

109         // 获取显示域的el
110         _getInputEl : function() {
111         	var self = this;
112         	var op = this.options;
113         	if (!op._inputEl) {
114         		op._inputEl = self.element.next();
115         	}
116         	return op._inputEl;
117         },
118         
119         // 获取隐藏域的el
120         _getHiddenEl : function() {
121         	var self = this;
122         	var op = this.options;
123         	if (!op._hiddenEl) {
124         		op._hiddenEl = self.element;
125         	}
126         	return op._hiddenEl;
127         },
128         
129         // 对象销毁方法
130         destroy : function() {
131         	this.options._hiddenEl = null;
132         	this.options._inputEl = null;
133             $.Widget.prototype.destroy.call(this);
134         },

136         // 绑定事件
137         _bindEvent : function() {
138         	var self = this;
139         	// 绑定显示域事件
140             var inputEvent = self._getEvent('input');
141             this._getInputEl().bind(inputEvent);
142             
143             // 绑定隐藏域事件
144             var hiddenEl= self._getHiddenEl();
145             hiddenEl.bind("onValueChanged", function(event, v) {
146         		self._triggerValueChanged($(this).val());
147             });
148         },
149         
150         // 获取绑定的事件
151         _getEvent : function(type) {
152             var ME = this, options = ME.options, element = this.element;
153             var self = this;
154             var op = this.options;
155             var inputEl = self._getInputEl();
156             var hiddenEl = self._getHiddenEl();
157 
158         	var inputEvent = {
159                 keydown : function(e) {
160                     var keyCode = e.keyCode;
161                     // KeyCode (48~57)==(0~9) (96~105)==小键盘(0~9) (←↑→↓)=(37,38,39,40) (190,8)==(小数点,Backspace) 
162                     if (48 <= keyCode && keyCode <= 57 // (0-9)
163                     		|| 96 <= keyCode && keyCode <= 105 // 小键盘(0-9) 
164                     		|| keyCode == 190 || keyCode == 110 // 小数点 .和小键盘小数点
165                     		|| keyCode == 109 || keyCode == 189 // (减号和小键盘减号)
166                     ) {
167                     	if (keyCode == 190 || keyCode == 110) {
168                     		if (op.decimals < 1) {// 如果允许输入的小数位为0，则不能输入小数点
169                     			return false;
170                     		} else {
171                     			return true;
172                     		}
173                     	}
174                     	var curValue = inputEl.val();
175                     	var curLength = curValue.length + 1; // 当前值没有回显
176                     	var maxLength = self.MAX_LENGTH;
177                     	if (op.decimals >= 1) {
178                         	if (curValue.lastIndexOf('.') >= 0) {
179                         		curLength -= 1;
180                         	} else {
181                         		curLength += op.decimals; // 为输入的小数位数留余量
182                         	}
183                     	}
184                     	if (curLength > maxLength) {
185                     		return false;
186                     	}
187                     	return true;
188                     } else if (keyCode == 37 || keyCode == 38 || keyCode == 39 || keyCode == 40 // (←↑→↓)
189                     		|| keyCode == 8 || keyCode == 46 // (Backspace, del)
190                     		|| keyCode == 9 || keyCode == 13 || keyCode == 108 // (Tab, enter, 小键盘enter)
191                    		) {
192                     	return true;
193                     }
194                     return false;
195                 },
196                 focus : function(e) { // 将隐藏域的值拷贝到显示域中
197                 	self._getInputEl().val(self._getHiddenEl().val());
198                 },
199                 blur : function(e) { // 将显示域按照要求的格式拷贝到隐藏域，然后将显示域的值格式化
200             			var prevValue = self._getHiddenEl().val();
201                 	self._getHiddenEl().val(self._getInputEl().val());
202             			self._triggerValueChanged(prevValue);
203                 }
204             };
205             if (type == 'input') return inputEvent;
206         },

209         _num2num : function(s, n) { // 限定小数位数
210         	n = n >= 0 && n <= 20 ? n : 2;
211         	s = parseFloat((s + "").replace(/[^\d\.-]/g, "")).toFixed(n);
212         	return s;
213         },
214         
215         _num2money : function(s) {
216         	s = (s + "").split(".");
217         	var l = s[0].split("").reverse();
218         	var r = s[1];
219         	if (r) {
220         		r = "."+r;
221         	} else {
222         		r= "";
223         	}
224         	var t = "";
225         	for (var i = 0; i < l.length; i++) {
226         		t += l[i] + ((i + 1) % 3 == 0 && (i + 1) != l.length && l[(i + 1)] !="-" ? "," : "");
227         	}
228         	return t.split("").reverse().join("") + r;
229         },
230         
231         _money2num : function(s) {
232         	return parseFloat(s.replace(/[^\d\.-]/g, ""));
233         },
234 
235         _triggerValueChanged : function(prevValue) { // 隐藏被修改，则更新显示域
236         	var self = this;
237         	var hiddenEl = self._getHiddenEl();
238         	var inputEl = self._getInputEl();
239         	var op = self.options;
240         	var value = hiddenEl.val();
241         	var realValue = self._num2num(value, op.decimals);
242         	if (!isNaN(realValue)) {
243         		if (realValue != value) { // 修正设置的值
244         			hiddenEl.val(realValue);
245         		}
246         		if (true == op['moneyFormat']) {
247             		var moneyValue = self._num2money(realValue);
248             		inputEl.val(moneyValue);
249         		} else {
250         			inputEl.val(realValue);
251         		}
252         	} else { // 如果输入的不是合法的数字，则清空输入框
253         		inputEl.val("");
254         		hiddenEl.val("");
255         	}
256         	if ($.isFunction(op.afterChange)) {
257         		var value = hiddenEl.val();
258         		if (prevValue !== value) {
259         			op.afterChange(value, prevValue);
260         		}
261         	}
262         },

275         },
