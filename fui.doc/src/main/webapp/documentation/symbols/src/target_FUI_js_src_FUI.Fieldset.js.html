1 /**
  2  * @name FFieldset
  3  * @class 
  4  * 字段集容器，和HTML的<fieldset>标签的功能类似；在一个form中，对组件进行分组和管理，浏览器会以特殊方式来显示它们，它们可能有特殊的边界，或者甚至可创建一个子表单来处理这些元素；可以折叠和展开。
  5  */
  6 
  7 /**@lends FFieldset# */
  8 
  9 /**
 10  * 标识(仅标签使用)
 11  * @name FFieldset#<ins>id</ins>
 12  * @type String
 13  * @default 随机生成
 14  * @example
 15  * 无
 16  */
 17 
 18 /**
 19  * 是否隐藏(仅标签使用)
 20  * @name FFieldset#<ins>visible</ins>
 21  * @type boolean
 22  * @default true
 23  * @example
 24  * 无
 25  */
 26 
 27 /**
 28  * 头部显示的标题文字。(仅标签使用)
 29  * @name FFieldset#<ins>title</ins>
 30  * @type String 
 31  * @default ""
 32  * @example
 33  * 无 
 34  */
 35 
 36 /**
 37  * 如果为true，FFieldset是可收缩的，并且有一个收起/展开按钮自动被渲染到它的头部工具区域，否则没有按钮(默认值为false)。(仅标签使用)
 38  * @name FFieldset#<ins>collapsible</ins>
 39  * @type Boolean
 40  * @default true
 41  * @example
 42  * 无 
 43  */
 44 
 45 /**
 46  * 如果为true，将FFieldset渲染成收缩的，否则渲染成展开的(默认值为false)。(仅标签使用)<br/>
 47  * 如果设置了此值为true，并且同时设置了collapsible为false，那么此值设置无效，<br/>
 48  * 请慎重设置此值：如果在组件初始化设置了此值为true，那么在Fieldset第一次展开的时候在部分浏览器下会造成组件的宽（高）度有调整，但是之后将会表现正常。
 49  * @name FFieldset#<ins>collapsed</ins>
 50  * @type Boolean
 51  * @default false 
 52  * @example
 53  * 无 
 54  */
 55 
 56 (function($) {
 57 	$.registerWidgetEvent("onExpand,onCollapse");
 58 	$.widget("FUI.FFieldset", {
 59 	    options : {
 60 	        /**
 61 	         * 此组件的整体宽度（包括border和padding，单位象素），默认为"auto"。<br/>
 62 	         * 支持数字形式的字符串或者带px单位的数字字符串，比如"40"、"40px"，都表示40个像素，字符串如"px50"、"abcpx"等其他非法形式的字符串则被忽略；
 63 	         * width也支持百分比，比如"10%"，则表示占用外层容器(div,td等)宽度的比率，此时请慎重设置左右margin、border和padding，否则会造成在不同浏览器下表现不一致不准确。<br/>
 64 	         * @name FPanel#width
 65 	         * @type String
 66 	         * @default "auto"
 67 	         * @example
 68 	         * 无
 69 	         */
 70 	        width : "auto",
 71 	        /**
 72 	         * 此组件的整体高度（包括border和padding，单位像素），默认为auto。<br/>
 73 	         * 仅支持数字形式的字符串或者带px单位的数字字符串，比如"40"、"40px"，都是表示40个像素；其他非法形式的字符串则被忽略。<br/>
 74 	         * 请慎重设置组件的margin的top和bottom值，在不同浏览器下可能是不同的显示效果。
 75 	         * height属性不支持百分比。
 76 	         * @name FPanel#height
 77 	         * @type String
 78 	         * @default "auto"
 79 	         * @example
 80 	         * 无
 81 	         */
 82 	        height : "auto",
 83 	        /**
 84 	         * 如果设置为true，表示控件在必要时自动显示横向或纵向滚动条。默认值为true。  (仅标签使用)
 85 	         * @name FFieldset#<ins>autoScroll</ins>
 86 	         * @type Boolean
 87 	         * @default true 
 88 	         * @example
 89 	         * 无 
 90 	         */
 91 	        autoscroll : true,
 92 	        /**
 93 	         * 收起组件后触发的函数
 94 	         * @event
 95 	         * @name FFieldset#onCollapse
 96 	         * @example
 97 	         * onCollapse : function(){
 98 	         *    //do something
 99 	         * }
100 	         */
101 	        onCollapse : null,
102 	        /**
103 	         * 展开组件后触发的函数
104 	         * @event
105 	         * @name FFieldset#onExpand
106 	         * @example
107 	         * onExpand : function(){
108 	         *    //do something
109 	         * }
110 	         */
111 	        onExpand : null,
112 
113 	        // 是否收起来
114 	        _isCollapsed : false,
115 	        // 是否需要在展开的时候重新计算高宽
116 	        _isNeedResize : false,
117 	        // 缓存jQuery对象：头部对象
118 	        _objHeader : null,
119 	        // 缓存jQuery对象：展开、收缩按钮
120 	        _objHeaderToggle : null,
121 	        // 缓存jquery对象：header
122 	        _objHeaderText : null,
123 	        // 缓存jQuery对象：bodyWrapper
124 	        _objBodyWrapper : null,
125 	        // 缓存body与外框相差的垂直像素个数
126 	        _bodyWrapperHGaps : 38,
127 	        // 缓存body与外框相差的水平像素个数
128 	        _bodyWrapperWGaps : 28
129 	    },
130 	    _create : function() {
131 		    // 初始化ID
132 		    var ID = this.element.attr("id");
133 		    var selfEl = this.element;
134 		    var op = this.options;
135 
136 		    op.id = ID;
137 		    // 初始化缓存对象
138 		    op._objHeader = $I(ID + "-legend");
139 		    op._objHeaderToggle = $I(ID + "-toggle");
140 		    op._objHeaderToggle.css("float", "left");
141 		    op._objHeaderToggle.css("margin", "2px");
142 		    op._objHeaderText = $I(ID + "-title");
143 		    op._objBodyWrapper = $I(ID + "-bwrap");
144 		    // 是否折叠
145 		    op._isCollapsed = this.element.hasClass("f-collapsed");
146 		    op._isNeedResize = op._isCollapsed;
147 		    if (!op._isNeedResize) {
148 			    op._bodyWrapperHGaps = selfEl.outerHeight(false) - op._objBodyWrapper.height();
149 			    op._bodyWrapperWGaps = selfEl.outerWidth(false) - op._objBodyWrapper.width();
150 		    }
151 		    // 绑定事件
152 		    this._bindEvent();
153 	    },
154 	    _init : function() {
155 		    // 初始化
156 		    this.setSize(this.options.width, this.options.height);
157 	    },
158 	    _bindEvent : function() {
159 		    var self = this;
160 		    var toggleEl = this.options._objHeaderToggle;
161 
162 		    // toggle绑定hover事件
163 		    toggleEl.hover(function(event) {
164 			    toggleEl.addClass("f-tool-toggle-over");
165 		    }, function(event) {
166 			    toggleEl.removeClass("f-tool-toggle-over");
167 		    });
168 
169 		    // 绑定点击事件
170 		    toggleEl.click(function() {
171 			    if (self.options._isCollapsed) {
172 				    self.expand();
173 			    } else {
174 				    self.collapse();
175 			    }
176 			    return false;
177 		    });
178 	    },
179 	    /**
180 	     * 销毁对象
181 	     * @name FFieldset#destroy
182 	     * @function
183 	     * @example
184 	     */
185 	    destroy : function() {
186 		    // 取消绑定的事件
187 		    this.options._objHeaderToggle.unbind();
188 		    // 销毁jQuery对象
189 		    this.options._objHeader = null;
190 		    this.options._objHeaderToggle = null;
191 		    this.options._objHeaderText = null;
192 		    this.options._objBodyWrapper = null;
193 		    // 调用父类的销毁方法
194 		    $.Widget.prototype.destroy.call(this);
195 	    },
196 	    /**
197 	     * 展开组件
198 	     * @name FFieldset#expand
199 	     * @function
200 	     * @example
201 	     */
202 	    expand : function() {
203 		    var self = this;
204 		    var selfEl = this.element;
205 		    var op = self.options;
206 		    if (op._isCollapsed) {
207 			    op._objBodyWrapper.slideDown("fast", function() {
208 				    self.element.addClass("f-collapsible").removeClass("f-collapsed");
209 				    op._isCollapsed = false;
210 				    if (op._isNeedResize) {
211 					    op._bodyWrapperHGaps = selfEl.outerHeight(false)
212 					            - op._objBodyWrapper.height();
213 					    op._bodyWrapperWGaps = selfEl.outerWidth(false)
214 					            - op._objBodyWrapper.width();
215 					    self.setSize(op.width, op.height);
216 					    op._isNeedResize = false;
217 				    }
218 				    if (op.autoscroll) {
219 					    op._objBodyWrapper.css("overflow", "auto");
220 				    }
221 				    // 调用回调或者触发事件
222 				    var canGo = true;
223 				    if ($.isFunction(op.onExpand)) {
224 					    var result = op.onExpand.call(self); // 回调方法返回false，则阻止触发事件
225 					    canGo = (result === false) ? false : true;
226 				    }
227 				    if (canGo) {
228 					    self.element.triggerHandler('onExpand');
229 				    }
230 				    $(this).css({
231 				    	zoom : 1
232 				    });
233 			    });
234 		    }
235 	    },
236 	    /**
237 	     * 收起组件
238 	     * @name FFieldset#collapse
239 	     * @function
240 	     * @example
241 	     */
242 	    collapse : function() {
243 		    var self = this;
244 		    var op = self.options;
245 		    if (!op._isCollapsed) {
246 			    op._objBodyWrapper.slideUp("fast", function() {
247 				    self.element.removeClass("f-collapsible").addClass("f-collapsed");
248 				    op._isCollapsed = true;
249 				    // 调用回调或者触发事件
250 				    var canGo = true;
251 				    if ($.isFunction(op.onCollapse)) {
252 					    var result = op.onCollapse.call(self);
253 					    canGo = (result === false) ? false : true;
254 				    }
255 				    if (canGo) {
256 					    self.element.triggerHandler('onCollapse');
257 				    }
258 			    });
259 		    }
260 	    },
261 	    /**
262 	     * 设置标题。 参数: title : String 被设置的title文本 返回值: void
263 	     * @name FFieldset#setTitle
264 	     * @function
265 	     * @param title String
266 	     * @return void
267 	     * @example
268 	     */
269 	    setTitle : function(title) {
270 		    if (title) {
271 			    this.options.title = title;
272 			    this.options._objHeaderText.text(title);
273 		    }
274 	    },
275 	    /**
276 	     * 设置组件的高宽，组件会自动调整内部toolbar的大小和位置
277 	     * @name FPanel#setSize
278 	     * @function
279 	     * @param w 组件的宽度，必须是数字或者数字类的字符串，或者百分比
280 	     * @param h 组件的高度，必须是数字或者数字类的字符串
281 	     * @example
282 	     */
283 	    setSize : function(w, h) {
284 		    var UTILS = window["$Utils"];
285 		    var op = this.options;
286 		    var selfEl = this.element;
287 		    var bodyEl = op._objBodyWrapper;
288 		    var bodyContent = bodyEl.children(".f-panel-body-content");
289 		    var contentSize = bodyContent.size();
290 
291 		    if (w) {
292 			    // border、padding占用的大小
293 			    var percentage = UTILS.getPercentage(w);
294 			    if (percentage) {
295 				    selfEl.get(0).style.width = percentage;
296 				    bodyEl.get(0).style.width = "auto";
297 				    if (contentSize === 1) {
298 					    bodyContent.get(0).style.width = "auto";
299 				    }
300 			    } else {
301 				    // 对于'40'和'40px'等同样处理为40
302 				    var gaps = op._bodyWrapperWGaps;
303 				    // var gaps = 0;
304 				    w = parseInt(w);
305 				    var realSize = w - gaps;
306 				    if (!isNaN(realSize)) {
307 					    if (realSize < 0) {
308 						    realSize = 0;
309 					    }
310 					    selfEl.get(0).style.width = realSize + "px";
311 					    if (contentSize === 1) {
312 						    bodyContent.get(0).style.width = realSize + "px";
313 					    } else {
314 						    bodyEl.get(0).style.width = realSize + "px";
315 					    }
316 				    }
317 			    }
318 		    }
319 		    if (h) {
320 			    // border、padding占用的大小
321 			    var percentage = UTILS.getPercentage(h);
322 			    if (percentage) {
323 				    selfEl.get(0).style.height = percentage;
324 				    bodyEl.get(0).style.height = "auto";
325 				    if (contentSize === 1) {
326 					    bodyContent.get(0).style.height = "auto";
327 				    }
328 			    } else {
329 				    // 对于'40'和'40px'等同样处理为40
330 				    var gaps = op._bodyWrapperHGaps;
331 				    var realSize = parseInt(h) - gaps;
332 				    if (!isNaN(realSize)) {
333 					    if (realSize < 0) {
334 						    realSize = 0;
335 					    }
336 					    if (contentSize === 1) {
337 						    var otherHeight = 0;
338 						    bodyContent.siblings().not('script').each(function() {
339 							    // alert($(this).attr('id') + ":" + $(this).outerHeight(true));
340 							    otherHeight += $(this).outerHeight(true);
341 						    });
342 						    realSize = (realSize - otherHeight) < 0 ? 0 : (realSize - otherHeight);
343 						    bodyContent.get(0).style.height = realSize + "px";
344 					    } else {
345 						    bodyEl.get(0).style.height = realSize + "px";
346 					    }
347 					    selfEl.get(0).style.height = "auto";
348 				    }
349 			    }
350 		    }
351 	    }
352 	});
353 })(jQuery);
