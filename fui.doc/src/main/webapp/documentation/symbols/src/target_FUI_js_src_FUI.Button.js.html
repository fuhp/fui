175 	         * 请慎重设置组件的margin的top和bottom值，在不同浏览器下可能是不同的显示效果。请不要设置button的padding、border的样式，否则会造成宽度计算不准确

187 	         * width也支持百分比，比如"10%"，则表示占用外层容器(div,td等)宽度的比率，此时请慎重button设置左右margin、border和padding，否则会造成在不同浏览器下表现不一致不准确。<br/>
188 	         * 请不要设置button的padding、border的样式，否则会造成宽度计算不准确

208 	    _create : function() {
209 		    // 初始化ID
210 		    this.options.id = this.element.attr("id");
211 		    this.options._disabled = this._getBoxEl().hasClass("f-state-disabled");
212

214 		    this._bindEvent();

218 	    },

220 	    _getBoxEl : function() {
221 	    	var op = this.options;
222 	    	var boxEl = op._objBoxEl;
223 	    	if (!boxEl) {
224 	    		boxEl = op._objBoxEl = this._getButtonEl().parent().parent();
225 	    	}
226 	    	return boxEl;
227 	    },
228 	    
229 	    _getButtonEl : function() {
230 	    	return this.element;
231 	    },
232 
233 	    _bindEvent : function() {
234 		    var self = this;
235 		    var op = self.options;
236 		    var boxEl = self._getBoxEl();
237 		    var bntEl = self._getButtonEl();
238 
239 		    boxEl.hover(function() {
240 		    	if (self.disabled()) {
241 		    		return;
242 		    	}
243 		    	boxEl.addClass("f-state-hover");
244 			}, function() {
245 		    	if (self.disabled()) {
246 		    		return;
247 		    	}
248 				boxEl.removeClass("f-state-hover f-state-active");
249 			}).mousedown(function() {
250 		    	if (self.disabled()) {
251 		    		return;
252 		    	}
253 				boxEl.addClass("f-state-active");
254 			}).mouseup(function() {
255 		    	if (self.disabled()) {
256 		    		return;
257 		    	}
258 				boxEl.removeClass("f-state-active");
259 			}).click(function() {
260 				self.click();
261 			});
262 			
263 		    bntEl.focus(function() {
264 		    	if (self.disabled()) {
265 		    		return;
266 		    	}
267 		    	boxEl.addClass("f-state-focus");
268 			}).blur(function() {
269 		    	if (self.disabled()) {
270 		    		return;
271 		    	}
272 				boxEl.removeClass("f-state-focus");
273 			});
274 	    },
275 	    
276 	    // 对象销毁方法
277 	    destroy : function() {
278 		    // 取消绑定的事件
279 	    	this._getBoxEl().unbind();
280 	    	this._getButtonEl().unbind();
281 	    	this.options._objBoxEl = null;
282 		    // 调用父类的销毁方法
283 		    $.Widget.prototype.destroy.call(this);
284 	    },

287 	     * 触发点击事件"onClick"。
288 	     * @name FSimpleButton#click
289 	     * @function
290 	     * @example
291 	     * $('#btn').FButton('click');
292 	     */
293 	    click : function() {
294 		    var self = this;
295 		    var op = self.options;
296 		    if (!self.disabled()) {
297 			    // 如果有注册的点击事件，则先调用注册的点击事件
298 			    var canGo = true;
299 			    if ($.isFunction(op.onClick)) {
300 				    // setTimeout(function() {
301 				    var result = op.onClick.call(self);
302 				    canGo = (result === false) ? false : true;
303 				    // }, 0);
304 			    }
305 			    // 触发其他注册方式的点击事件
306 			    if (canGo) {
307 				    this.element.triggerHandler("onClick");
308 			    }
309 		    }
310 	    },
311 	    
312 	    /**
313 	     * 禁用/启用组件。
314 	     * @name FSimpleButton#disabled
315 	     * @function
316 	     * @param disabled 如果为false，则表示启用；否则则表示禁用
317 	     * @example
318 	     * $('#btn').FSimpleButton('disabled');
319 	     */
320 	    disabled : function(disabled) {
321 		    var op = this.options;
322 		    if (arguments.length === 0) {
323 		    	return op._disabled;
324 		    }
325 		    var boxEl = this._getBoxEl();
326 		    var bntEl = this._getButtonEl();
327 		    if (disabled === false) {
328 			    op._disabled = false;
329 			    boxEl.removeClass("f-state-hover f-state-active f-state-focus f-state-disabled");
330 			    bntEl.attr("disabled", false);
331 		    } else {
332 			    op._disabled = true;
333 			    boxEl.removeClass("f-state-hover f-state-active");
334 			    boxEl.addClass("f-state-disabled");
335 			    bntEl.attr("disabled", true);
336 		    }
337 	    },

346 	     */
347 	    setText : function(text) {
348 		    if (text) {
349 			    this._getButtonEl().html(text);
350 		    } else {
351 			    this._getButtonEl().html(" ");
352 		    }
