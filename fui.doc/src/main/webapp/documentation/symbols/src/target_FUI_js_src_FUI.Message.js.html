281 	    /**
282 	     * 弹出指定参数的消息框
283 	     * @name FMessage#show
284 	     * @function
285 	     * @param op 消息框的弹出选项，可以有下列属性：
286 	     * <ol>
287 	     * <li><b>title:</b> 消息框标题；</li>
288 	     * <li><b>msg:</b> 消息的内容，对于prompt，就是输入框的提示信息；</li>
289 	     * <li><b>callback:</b> 输入框在关闭的时候出发的回调方法，非必须；</li>
290 	     * <li><b>width:</b> 输入框的宽度，默认情况下自动设置合适的宽度，也可以通过设置此属性修改；</li>
291 	     * <li><b>icon:</b> msg前的图标，目前只支持 $.FMessage.INFO、$.FMessage.QUESTION、
292 	     * $.FMessage.ERROR、$.FMessage.WARNING；如果没有设置此属性，则不显示图标；</li>
293 	     * <li><b>prompt:</b> boolean类型，true则表示显示一个带输入框的消息框，此时icon属性无效；</li>
294 	     * <li><b>buttons:</b> Message显示的按钮
295 	     * <pre>
296 	     * $.FMessage.OK 只显示一个"OK"按钮
297 	     * $.FMessage.OKCANCEL 显示一个"OK"按钮，一个"Cancel"按钮
298 	     * $.FMessage.YES 只显示一个"Yes"按钮
299 	     * $.FMessage.NO 只显示一个"No"按钮
300 	     * $.FMessage.YESNO 只显示一个"Yes"按钮，一个"No"按钮
301 	     * $.FMessage.CANCEL 只显示一个"Cancel"按钮
302 	     * $.FMessage.YESNOCANCEL 显示"Yes"、"No"、"Cancel"按钮
303 	     * 
304 	     * 这些按钮可以自由组合，比如 $.FMessage.OK|$.FMessage.CANCEL 效果与 $.FMessage.OKCANCEL一致
305 	     * </pre>
306 	     * </li>
307 	     * <li><b>buttonTexts:</b> 配置按钮对应文字
308 	     * <pre>
309 	     *  $.FMessage.show({
310 	     *    title: 'Icon Support',
311 	     *    msg: 'Here is a message with an icon!',
312 	     *    buttons:  $.FMessage.OK | $.FMessage.YESNOCANCEL,
313 	     *    icon: $.FMessage.QUESTION,
314 	     *    buttonTexts : {
315 	     *      yes : "I say yes!",
316 	     *      no : "No more...",
317 	     *      cancel : "Bye."
318 	     *    }
319 	     *  });
320 	     * </pre>
321 	     * </li>
322 	     *  <li><b>closeIcon</b> 是否显示右上角的关闭按钮，false则不显示按钮，默认显示</li>
323 	     * </ol>
324 	     * @example
325 	     */
326 		show : function(op) {
327 			var self = this;
328 			var msgEl = self._getMessageEl(); // 必须在前面调用
329 			
330 			if (self._isShow) {
331 				self._cmdQueue.push(op);
332 				return;
333 			}
334 			self._isShow = true;
335 			
336 			// 修改标题
337 			var title = op.title || " ";
338 			self._getElement("title").html(title);
339 			// 重置按钮
340 			var bntCount = self._resetButtons(op.buttons, op.buttonTexts);
341 			
342 			// 调整message的大小
343 			var width = "345px";
344 			if (op.width) {
345 				width = op.width;
346 			} else {
347 				if (bntCount == 1) {
348 					width = "250px";
349 				} else if (bntCount == 2) {
350 					width = "280px";
351 				} else if (bntCount >= 3) {
352 					width = "345px";
353 				}
354 			}
355 			msgEl.width(width);
356 			
357 			// 重置body内容
358 			self._resetBody(op, width);
359 			
360 			// 是否有右上角的关闭按钮
361 			var iconEl = self._getElement("closeIcon");
362 			if (op.closeIcon === false) {
363 				iconEl.css("display", "none");
364 			} else {
365 				iconEl.css("display", "block");
366 			}
367 			
368 			// 显示message
369 			self._showMask();
370 			msgEl.FMessage("show", {
371 				show : {
372 					callback : function() {
373 						$(document).bind("keydown.FMessage", function(e) {
374 							if (e.which == 27) {
375 								self._bntClick("cancel");
376 							}
377 						});
378 					}
379 				},
380 				hide : {
381 					callback : function() {
382 						self._hideMask();
383 						$(document).unbind("keydown.FMessage");
384 						self._status.text = self._getElement("promptInput").val();
385 						if ($.isFunction(op.callback)) {
386 							op.callback(self._status.choise, self._status.text);
387 						}
388 						self._isShow = false;
389 						self._processQueue(); // 处理在弹出框过程中弹出来的框
390 					}
391 				}
392 			});
393 		},
394 		
395 		_processQueue : function() {

398 			if (queue.length == 0) {
399 				return;
400 			}
401 			self.show(queue.shift());
402 		},

418 		_resetBody : function(op, width) {
419 			width = parseInt(width);
420 			var self = this;
421 			var tipIcon = op.icon;
422 			var msg = op.msg || " ";
423 			var contentEl = self._getElement("content");
424 			var iconContentEl = self._getElement("iconContent");
425 			var promptEl = self._getElement("prompt");
426 			if (op.prompt == true) {
427 				contentEl.css("display", "none");
428 				promptEl.css("display", "block");
429 				iconContentEl.css("display", "none");
430 				self._getElement("promptMsg").html(msg);
431 				self._getElement("promptInput").val("");
432 			} else if (tipIcon) { // 有icon的内容
433 				var icons = [self.INFO, self.QUESTION, self.WARNING, self.ERROR];
434 				contentEl.css("display", "none");
435 				promptEl.css("display", "none");
436 				iconContentEl.css("display", "block");
437 				// 设置图标
438 				var iconEl = self._getElement("tipIcon");
439 				iconEl.removeClass(icons.join(" "));
440 				iconEl.addClass(tipIcon);
441 				// 设置内容
442 				var content = self._getElement("tipContent");
443 				// add 20121227 hanyin 修复icon无法隐藏的问题：将table布局改为float
444 				if (!isNaN(width)) {
445 					content.width(width-20-50);
446 				}
447 				// end add 
448 				content.html(msg);
449 			} else {
450 				contentEl.css("display", "block");
451 				promptEl.css("display", "none");
452 				iconContentEl.css("display", "none");
453 				self._getElement("content").html(msg);
454 			}
455 		},

457 		_resetButtons : function(buttons, buttonTexts) {
458 			var self = this;
459 			buttons = buttons || 0;
460 			if (buttons == 0) {
461 				var el = self._getElement("buttonGroup");
462 				el.css("display", "none");
463 				return 0;
464 			} else {
465 				buttonTexts = buttonTexts || {};
466 				self._getElement("buttonGroup").css("display", "block");
467 				var iconCount = 0;
468 				$.each(["ok", "yes", "no", "cancel"], function(i, name) {
469 					var el = self._getElement(name+"Bnt-wrapper");
470 					if (buttons & self[name.toUpperCase()]) {
471 						if (buttonTexts[name]) {
472 							self._getElement(name+"Bnt").text(buttonTexts[name]);
473 						}
474 						el.parent().show();
475 						iconCount ++ ;
476 					} else {
477 						el.parent().hide();
478 					}
479 				});
480 				return iconCount;
481 			}
482 		},
