274         _nodeClickHandler : function(target) {
275             var ME = this,UTILS = window['$Utils'];
276             //树节点前方的收缩折叠按钮，单击展开或者收缩树节点，触发加载数据
277             var nodeData = ME._getNodeDataByDom(target);
278             if (nodeData.leaf == true || nodeData.leaf == "true") {
279             	return;
280             }
281             var liEl = $(target).parents("li");
282             var nodeStatus = liEl.attr('f_status');
283             var ulId = nodeData.ulId;
284             if (ulId) {
285                 //已经生成子节点了
286                 if ('close' == nodeStatus) {
287                     //展开树节点
288                     ME._expandNode(nodeData);
289 
290                 } else if ('open' == nodeStatus) {
291                     //隐藏树节点
292                     ME._collapseNode(nodeData);
293                 }
294             } else {
295                 if (ME.options.syncLoad) {
296                     //同步加载
297                     if (ME.hasLoaded) {
298                         var children = nodeData.children;
299                         if (children && children.length) {
300                             //存在子节点
301                             ME._changeIcons(nodeData, '-close', '-open');
302                             ME._renderNode(nodeData, children);
303                         } else {
304                             //不存在子节点
305                             //改变节点
306                             ME._clearPlusIcons(nodeData);
307                         }
308                     } else {
309                         //首次加载数据
310                         ME._loadNode(nodeData);
311                     }
312                 } else {
313                     //异步加载
314                     ME._loadNode(nodeData);
315                 }
316             }
317         },

405         _adjustParentNode : function(nodeData) {
406             var selectModel = this.options.selectModel;
407             //根据nodeData中的pid，查找该节点的父节点对象。
408             var pid = nodeData.pid;
409             if (pid) {
410                 var parentNode = this.dataStore[pid];
411                 if (!parentNode) {
412                     return;
413                 }
414                 var checkboxId = parentNode.checkboxId;
415                 var checkboxEl = $I(checkboxId);
416                 var checkboxDom = checkboxEl.get(0);
417                 //如果父节点选中，则需要判断父节点的子节点是否全部选中
418                 if ('selectChildren' == selectModel) {
419                     var children = parentNode.children;
420                     if (!children || !$.isArray(children)) {
421                         return;
422                     }
423                     var length = children.length;
424                     var isAllChecked = true;
425                     var isAllNoChecked = true;
426                     for (var i = 0; i < length; i++) {
427                         var node = children[i];
428                         if (!node.checked) {
429                             isAllChecked = false;
430                         } else {
431                             isAllNoChecked = false;
432                         }
433                     }
434                     if (parentNode.checked) {
435                         if (isAllChecked) {
436                             // 改变样式 ,正常选中图标
437                             checkboxDom.className = this.nodeSelectedCls;
438                         } else {
439                             checkboxDom.className = this.nodeSelectedPartCls;
440                         }
441                     } else {
442                         if (isAllNoChecked) {
443                             checkboxDom.className = this.nodeUnSelectedCls;
444                         } else {
445                             checkboxDom.className = this.nodeUnSelectedPartCls;
446                         }
447                     }
448                 } else if ('selectParent' == selectModel) {
449                     var pChildren = parentNode.children;
450                     var length = pChildren.length;
451                     var isAllChecked = true;
452                     var isAllNoChecked = true;
453                     for (var i = 0; i < length; i++) {
454                         var node = pChildren[i];
455                         if (!node.checked) {
456                             isAllChecked = false;
457                         } else {
458                             isAllNoChecked = false;
459                         }
460                     }
461                     if (nodeData.checked) {
462                         //判断parentNode的子节点是否全部选中。如果相邻子节点全部选中，则全部选中。如果不是全部选中，则半选中。
463                         if (isAllChecked) {
464                             parentNode.checked = true;
465                             checkboxDom.className = this.nodeSelectedCls;
466                         } else {
467                             parentNode.checked = true;
468                             checkboxDom.className = this.nodeSelectedPartCls;
469                         }
470                         this.selectedNodes[parentNode.id] = parentNode;
471                     } else {
472                         //判断parentNode的子节点是否有选中的，如果有选中，则半选中，否则，不选中。
473                         if (isAllNoChecked) {
474                             parentNode.checked = false;
475                             delete this.selectedNodes[parentNode.id];
476                             checkboxDom.className = this.nodeUnSelectedCls;
477                         } else {
478                             parentNode.checked = true;
479                             this.selectedNodes[parentNode.id] = parentNode;
480                             checkboxDom.className = this.nodeSelectedPartCls;
481                         }
482                     }
483                 }
484                 parentNode.pid && this._adjustParentNode(parentNode);
485             }
486         },

492                 return;
493             }
494             var selectModel = this.options.selectModel;
495             if ('selectChildren' == selectModel) {
496                 //选中其下所有子节点
497                 var length = children.length;
498                 for (var i = 0; i < length; i++) {
499                     var node = children[i];
500                     var checkboxId = node.checkboxId;
501                     if (checkboxId) {
502                         var checkboxEl = $I(checkboxId);
503                         var checkboxDom = checkboxEl.get(0);
504                         if (nodeData.checked) {
505                             node.checked = true;
506                             this.selectedNodes[node.id] = node;
507                             checkboxDom.className = this.nodeSelectedCls;
508                         } else {
509                             node.checked = false;
510                             delete this.selectedNodes[node.id];
511                             checkboxDom.className = this.nodeUnSelectedCls;
512                         }
513                         node.children && this._adjustChildrenNode(node);
514                     }
515                 }
516             }
