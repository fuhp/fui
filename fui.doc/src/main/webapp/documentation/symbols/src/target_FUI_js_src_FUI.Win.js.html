402          * 重新计算window的位置，需要参数：组件的宽度，组件的高度。

406         _rePosition : function(w, h) {
407             //计算win组件展现的位置
408             var options = this.options;
409             w = w || options.width,h = h || options.height;
410             var windowEl = $(window),documentEl = $(document);
411             var windowW = windowEl.width(),windowH = windowEl.height();
412             var isBoxModel = jQuery.support.boxModel;
413             var scrollLeft = isBoxModel && document.documentElement.scrollLeft || document.body.scrollLeft;
414             var scrollTop = isBoxModel && document.documentElement.scrollTop || document.body.scrollTop;
415             var left = (windowW - w) / 2 + scrollLeft;
416             var top = (windowH - h) / 2 + scrollTop;
417             top < 0 && (top = 0),left < 0 && (left = 0);
418             this._showAt(left, top);
419         },
