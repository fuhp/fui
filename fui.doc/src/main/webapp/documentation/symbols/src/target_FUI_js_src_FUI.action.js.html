 14 //拖拽插件
 15 $.Dragable = function(targetId, widgetId) {
 16     this.targetId = targetId;
 17     this.widgetId = widgetId;
 18     this._init();
 19     this._bindEvent();
 20 
 21 }
 22 
 23 $.Dragable.prototype = {
 24 
 25     destroy : function() {
 26 
 27     },
 28 
 29     _init : function() {
 30         this.targetEl = $('#' + this.targetId);
 31         this.widgetEl = $('#' + this.widgetId);
 32         this.mousePos = {};
 33     },
 34 
 35     _bindEvent : function() {
 36         this.targetEl.mousedown($.proxy(this.onMousedown, this));
 37     },
 38 
 39     onMousedown : function(e) {
 40 
 41         var target = e.target;
 42         var className = target.className;
 43         if (className.indexOf('f-win-close') == -1 && className.indexOf('f-win-max') == -1 && className.indexOf('f-win-restore') == -1) {
 44             var w = this.widgetEl.outerWidth(true);
 45             var h = this.widgetEl.outerHeight(true);
 46             var position = this.widgetEl.position();
 47             var documentEl = $(document);
 48             var bodyEl = $('body');
 49             bodyEl.closeSelect();
 50             //生成对应的虚框
 51             var htmlArr = [];
 52              htmlArr.push('<div id="dottedLine" class="f-dotted-line" style="');
 53             // begin 20121213 hanyin 增加zIndex的设置
 54             var zIndex = parseInt(this.widgetEl.css("zIndex"));
 55             if (!isNaN(zIndex)) {
 56             	htmlArr.push('z-index:' + (zIndex+1)+";");
 57             }
 58             // end 20121213 hanyin 增加zIndex的设置
 59             htmlArr.push('width:' + w + 'px;');
 60             htmlArr.push('height:' + h + 'px;');
 61             htmlArr.push('top:' + position.top + 'px;');
 62             htmlArr.push('left:' + position.left + 'px;');
 63             htmlArr.push('"></div>');
 64             bodyEl.append(htmlArr.join(''));
 65 
 66             this.dottedLineEl = $('#dottedLine');
 67             var pageX = e.pageX,pageY = e.pageY;
 68             this.mousePos = {pageX:pageX,pageY:pageY};
 69             documentEl.mousemove($.proxy(this.onMousemove, this));
 70             documentEl.one('mouseup', $.proxy(this.onMouseup, this));
 71         }
 72     },
 73 
 74     onMousemove : function(e) {
 75         var pageX = e.pageX ,pageY = e.pageY ,dottedLineEl = this.dottedLineEl,mousePos = this.mousePos;
 76         var ox = pageX - mousePos.pageX,oy = pageY - mousePos.pageY;
 77 
 78         var position = dottedLineEl.position();
 79         var top = position.top + oy ;
 80         var left =  position.left + ox ;
 81         if(top<=0){
 82             top = 0;
 83             //{ 20130409 hanyin 避免鼠标超出窗口的情况下报js错的问题
 84             $(document).unbind('mouseup');
 85             this.onMouseup();
 86             //} 20130409 hanyin
 87         }
 88         var topPx = top + "px",leftPx = left+ 'px';
 89 
 90         var dottedLineStyle = dottedLineEl.get(0).style;
 91         dottedLineStyle.top = topPx;
 92         dottedLineStyle.left = leftPx;
 93 
 94         this.mousePos = {pageX:pageX,pageY:pageY};
 95     },
 96 
 97     onMouseup : function() {
 98         var documentEl = $(document);
 99         var bodyEl = $('body');
100         var position = this.dottedLineEl.position();
101         var widgetStyle = this.widgetEl.get(0).style;
102 
103         widgetStyle.top = position.top + 'px';
104         widgetStyle.left = position.left + 'px';
105 
106         this.dottedLineEl.remove();
107         this.dottedLineEl = null;
108         bodyEl.openSelect();
109         documentEl.unbind('mousemove');
110     },
111 
112     start: function() {
113         this.targetEl.mousedown($.proxy(this.onMousedown, this));
114     },
115 
116     stop : function() {
117         this.targetEl.unbind('mousedown', $.proxy(this.onMousedown, this));
118     }
119 }
