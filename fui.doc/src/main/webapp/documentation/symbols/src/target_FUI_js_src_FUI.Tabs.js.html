397 	    _bindTabHeadEvent : function(el) {
398 	    	var self = this;
399 	    	var op = self.options;
400     		el
401 	    	/* 为了快，不绑定动画效果
402 	    	.hover(function(){
403 	    		if (!$(this).hasClass("f-tabs-active")) {

411 	    		var tabId = href.substring(href.lastIndexOf("#")+1);
412 	    		if (op._currentTabId == tabId) { // 如果 已经打开，则不重复打开，因此也不会触发事件
413 	    			//  20130129 修改 hanyin 页签上鼠标点击事件不返回false，避免右键事件被拦截
414 //	    			return false;
415 	    			return;
416 	    		}
417 	    		self.activate(tabId);
418     			//  20130129 修改 hanyin 页签上鼠标点击事件不返回false，避免右键事件被拦截
419 //    			return false;
420     			return;
421 	    	});
422     		
423     		// add 20130125 hanyin 绑定页签双击事件
424     		if (op.onTabDblClick) {
425     			el.dblclick(function() {
426     				var hid = $(this).attr("id");
427     				var index = op._tabsHeadMapping.indexOf(hid);
428     				var id = op._tabsBodyMapping.get(index);
429         			self._tryExcuteFunc(self, op.onTabDblClick, [index, id]);
430     			});
431     		}
432     		// end add 20130125 hanyin 绑定页签双击事件
433     		
434     		if (el.hasClass("f-tab-closable")) {
435     			var closeBnt = el.children().children(".f-tab-close-bnt:first");
436     			closeBnt.click(function(e) {
437     				e.stopPropagation(); // 防止对应的标签被激活

439     				return false;

445 	    	var op = this.options;
446 	    	var self = this;
447 	    	var ctlBox = self._calElementBox("header-ctl"); // 外框窗口
448 	    	var innerBox = self._calHeaderItemsBox(); // 内框滚动盒子
449 	    	var realScrollGap = 0;
450 	    	if (innerBox.left+op.scrollGap > ctlBox.left) {
451 	    		realScrollGap = ctlBox.left - innerBox.left;
452 	    	} else {
453 	    		realScrollGap = op.scrollGap;
454 	    	}
455 	    	var ulEl = self._getElement("header-ctl", self._getElement("header")).children();
456     		ulEl.css("margin-left", (parseInt(ulEl.css("margin-left")) + realScrollGap)+"px");
457     		// 重新调整按钮的状态
458     		self._adjustTabScroller();
