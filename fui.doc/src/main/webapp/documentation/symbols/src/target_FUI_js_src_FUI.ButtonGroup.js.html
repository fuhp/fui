  9 /**
 10  * 所有的button的排布：'center':表示居中，'left':居左，'right':居右。
 11  * @name FButtonGroup#<ins>buttonAlign</ins>
 12  * @type String
 13  * @default 'center'
 14  * @example
 15  * 无
 16  */
 17 
 18 (function($) {
 19 	$.FUI.FToolGroup = $.FUI.FToolGroup || {};
 20 	$.FUI.FToolGroup.generateHtml = function(op) {
 21 		var id = op.id || $Utils.genId("f-toolGroup"); // 0
 22 		var style = op.style || ""; // 1
 23 		var classes = op.classes || ""; // 2
 24 		var width = op.width || "auto"; // 3
 25 		var height = op.height || "auto"; // 4
 26 		var toolAlign = op.toolAlign || "center"; //5
 27 		var toolspacing = op.toolspacing || "2"; // 6
 28 		var toolpadding = op.toolpadding || "0"; // 7
 29 		var nested = ""; // 8
 30 		
 31 		function genItemHtml(item) {
 32 			var html = "<td class='f-tool-cell'>" + item + "</td>"; 
 33 			return html;
 34 		}
 35 		
 36 		var items = op.items;
 37 		for (var i=0; i<items.length; i++) {
 38 			nested += genItemHtml(items[i]);
 39 		}
 40 		
 41 		var template = "\
 42 			<div id='{0}' class='f-toolGroup f-tools-{5} {2}' style='{1}'> \
 43 				<table cellspacing='{6}' cellpadding='{7}' \
 44 						style='height:{4};width:{3};' class='f-toolGroup-ct'> \
 45 					<tbody><tr>{8}</tr> \
 46 					</tbody> \
 47 				</table> \
 48 			</div>\
 49 		";
 50 		return $Utils.format(template, id, style, classes, width, height, 
 51 				toolAlign, toolspacing, toolpadding, nested);
 52 	};
 53 
 54 	$.FUI.FButtonGroup = $.FUI.FButtonGroup || {};
 55 	$.FUI.FButtonGroup.generateHtml = function(op) {
 56 		op.id = op.id || $Utils.genId("f-buttonGroup");
 57 		op.classes = (op.classes || "") + " f-buttonGroup ";
 58 		op.toolAlign = op.buttonAlign;
 59 		return $.FUI.FToolGroup.generateHtml(op);
 60 	};
 61 	
 62 })(jQuery);
