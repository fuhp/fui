166 	    _tryExcuteFunc : function(obj, func, args) {
167 	    	var EMPTY_FUNC = function() {};
168 	    	var f = func || EMPTY_FUNC;
169 	    	if (!$.isFunction(func)) {
170 	    		try {
171 	    			f = eval(func);
172 	    			if (!$.isFunction(f)) {
173 	    				return;
174 	    			}
175 	    		} catch (e) {
176 	    			return;
177 				}
178 	    	}
179 	    	args = args || [];
180 	    	return f.apply(obj, args);

194 	    show : function(op) {
195 	    	if (this.options._isShow === true) {
196 	    		this.hide(); // 隐藏菜单之后会清空菜单的状态
197 	    	}
198 	    	var self = this;
199 	    	var selfEl = this.element;
200 	    	op = op || {};
201 	    	self.options._curOptions = op = self._processOptions(op);
202 	    	// 触发beforeShow事件
203 	    	self._tryExcuteFunc(self, op.beforeShow, [op]);
204 	    	
205 	    	// 解析静态数据
206 	    	if (self.options._isNewStaticData) {
207 	    		var idStack = [];
208 	    		var menuStack = [];
209 	    		var funcStack = [];
210 	    		var html = self._tryParseStaticData(self.options.id, op.staticData, idStack, menuStack, funcStack);
211 	    		if (html != null && html.length != 0) {
212 		    		// 删除动态生成的子菜单的DOM结构，避免DOM结构溢出
213 	    			self._destroySubMenus();
214 	    			// 保存当前动态生成的子菜单
215 	    			self.options._staticMenuIds = idStack;
216 	    			// 修改menu的内部元素
217 		    		selfEl.html(html);
218 		    		selfEl.data("init.FMenu.FUI", false); // 将menu置为未初始化的状态
219 		    		if (menuStack.length != 0) {
220 			    		// 插入菜单的DOM结构
221 			    		$("body").append(menuStack.join(""));
222 			    		// 执行初始化脚本
223 			    		for (var i=0; i<funcStack.length; i++) {
224 			    			funcStack[i]();
225 			    		}
226 		    		}
227 	    		}
228 	    		self.options._isNewStaticData = false;
229 	    	}
230 
231 	    	// 将参数进行拷贝一份，避免污染
232 	    	// 如果组件还没有初始化，则尝试初始化
233 	    	var hasInit = selfEl.data("init.FMenu.FUI");
234 	    	if (hasInit !== true) {
235 	    		this._initMenu(op);
236 	    		selfEl.data("init.FMenu.FUI", true);
237 	    	}
238 	    	this.options._popupBox.FPopupBox("show", op);
239 	    	this.options._isShow = true;
240 	    	setTimeout(function() { // 将绑定事件放到浏览器的事件队列中，保证菜单关闭的事件可以正常被注册
241 	    		var eventType = "click.FMenu-" + self.options.id;
242 		    	$(document).one(eventType, function() {
243 		    		self.hide();
244 		    	});
245 	    	}, 0);
246 	    	// 触发onShow事件
247 	    	self._tryExcuteFunc(self, op.onShow, [op]);
248 	    },

252 	    _tryParseStaticData : function(menuId, staticData, idStack, menuStack, funcStack) {
253 	    	if (staticData != null) {
254 	    		idStack = idStack || [];
255 	    		menuStack = menuStack || [];
256 	    		funcStack = funcStack || [];
257 	    		var jsonObj = staticData;
258 		    	if ($Utils.isString(jsonObj)) {
259 		    		try {
260 		    			jsonObj = eval(jsonObj);
261 		    		} catch (e) {
262 		    			jsonObj = "";
263 					}
264 		    	}
265 		    	if (!$Utils.isArray(jsonObj)) { // 不是合法的menu的json格式
266 		    		return null;
267 		    	}
268 		    	var ID = menuId;
269 		    	var html = [];
270 		    	// begin 20121214 hanyin 给menu增加 icon与文本之间的分隔线
271 		    	html.push("<li id='" + ID+ "-icon-seperator' class='f-menu-icon-seperator'></li>");
272 		    	// end 20121214 hanyin 给menu增加 icon与文本之间的分隔线
273 		    	for (var i=0; i<jsonObj.length; i++) {
274 		    		var item = jsonObj[i];
275 		    		item.id = item.id || (ID + "-gen-item-" + i);
276 		    		var subMenuData = item.children || [];
277 		    		item.subMenuId = null;
278 		    		if ($Utils.isArray(subMenuData) && subMenuData.length > 0) {
279 		    			item.subMenuId = ID + "-gen-subMenu-" + i;
280 		    			idStack.push(item.subMenuId);
281 			    		this._registerSubMenu(item.subMenuId, subMenuData, menuStack, funcStack);
282 		    		}
283 		    		this._generateMenuItemHTML(item, html);
284 		    	}
285 		    	return html.join("");
286 	    	}
287 	    	return null;

289 	    _generateMenuItemHTML : function(item, html) {
290 	    	var id = item.id;
291 	    	var text = (item.text || " ").length==0?(" "):item.text;
292 	    	var iconCls = item.iconCls || "";
293 	    	var disable = ((item.disable||"false")=="true")?true:false;
294 	    	var onClick = item.onClick || "";
295 	    	var url = item.url || "";
296 	    	var title = item.title || ""; // 20130108 add hanyin 增加title属性
297 	    	var checked = item.checked;
298 	    	var checkedClass = "";
299 	    	if (checked != null) {
300 	    		checked = checked.toString();
301 	    		if (checked == "true") {
302 	    			checkedClass = "f-state-checked";
303 	    		} else if (checked == "false") {
304 	    			checkedClass = "f-state-unchecked";
305 	    		}
306 	    	}
307 	    	if (text == "-") {
308 	    		html.push("<li class='f-menu-item f-menu-item-seperator'> </li>");
309 	    	} else {
310 	    		html.push("<li class='f-menu-item'>");
311 	    		html.push("<a id='" +id+ "'");
312 	    		html.push(" href='#" + (item.subMenuId != null?item.subMenuId:"") + "'");
313 	    		html.push(" clickEvent='" + onClick + "'");
314 	    		html.push(" url='" + url + "'");
315 	    		// add 20130108 hanyin 给menuItem增加title属性
316 	    		if (title) {
317 		    		html.push(" title=\"" + title.replace("\"", "'") + "\"");
318 	    		}
319 	    		// end add 20130108 hanyin 给menuItem增加title属性
320 	    		html.push(" class='f-corner-all " + (disable?"f-state-disabled":"") + " " + checkedClass + "'>");
321 	    		if (iconCls.length != 0) {
322 	    			html.push("<span class='f-icon " +iconCls+ "'></span>");
323 	    		}
324 	    		if (item.subMenuId != null) {
325 	    			html.push("<span class='f-menu-icon f-icon f-icon-item-arrow'></span>");
326 	    		}
327 	    		html.push(text);
328 	    		html.push("</a></li>");
329 	    	}

376 	    _adjustMenuWidth : function(minWidth, maxWidth) {
377 	    	var w = this.element.width();
378 	    	var resultW = w;
379 	    	var min = parseInt(minWidth);
380 	    	var max = parseInt(maxWidth);
381 	    	if (!isNaN(min)) {
382 	    		if (min < 50) {
383 	    			min = 50;
384 	    		}
385 	    		resultW = w<min?min:w;
386 	    	} else if (!isNaN(max) && w > max) {
387 	    		resultW = max;
388 	    	}
389 	    	if (resultW != w) {
390 	    		this.element.width(resultW);
391 	    	}
392 	    	return resultW;

395 	    _bindItemEvent : function(el, op) {
396 	    	var aEl = el;
397 	    	var self = this;
398 	    	var liEl = aEl.parent();
399 	    	var selfOp = this.options;
400 	    	aEl.hover(function() {
401 	    		if (aEl.hasClass("f-state-disabled")) {
402 	    			return;
403 	    		}
404 	    		if (self.options._objActiveItem != null 
405 	    				&& self.options._objActiveItem.attr("id") == aEl.attr("id")) {
406 	    			return; // 避免重复打开
407 	    		}
408 	    		// 尝试关闭其他子菜单
409 	    		self._tryHideSubMenu();
410 	    		aEl.addClass("f-state-focus");
411 	    		selfOp._objFocusItem = aEl;
412 	    		// 尝试展开子菜单
413 	    		self._tryShowSubMenu(aEl, op);
414 	    	}, function() {
415 	    		if (aEl.hasClass("f-state-disabled")) {
416 	    			return;
417 	    		}
418 	    		aEl.removeClass("f-state-focus");
419 	    		selfOp._objFocusItem = null;
420 	    		// 尝试关闭子菜单
421 	    	});
422 	    	var ME = this; // 这里很奇怪，在aEL的click方法里面self是未定义的，所以需要使用一个新的ME变量
423 	    	aEl.click(function(e) {
424 	    		// 如果菜单项无效则直接返回
425 	    		if (aEl.hasClass("f-state-disabled")) {
426 	    			return false;
427 	    		}
428 	    		// 触发菜单点击事件
429 	    		var item = {
430 	    				id: aEl.attr("id"), 
431 	    				url: aEl.attr("url"), 
432 	    				menuId: ME.options.id, 
433 	    				parentMenuId: (ME.options._curOptions || {})["_parentMenuId"],
434 	    				text: aEl.text()
435 	    		};
436 	    		if (ME._triggerItemClick(aEl.attr("clickEvent"), item, e) !== false) {
437 	    	    	ME.triggerMenuClick(item, e);
438 	    		}
439 	    		// 触发document的点击事件，关闭菜单
440 	    		$(document).trigger("click.FMenu-" + ME.getRootMenuId());
441 	    		// e.stopPropagation(); // 使用这个会触发事件多次
442 	    		e.stopImmediatePropagation();
443 	    		return false; // 阻止<a>默认行为
444 	    	});
445 	    	$("> span.f-icon-item-arrow", aEl).hover(function() {
446 	    		if (aEl.hasClass("f-state-disabled")) {
447 	    			return;
448 	    		}
449 		    	self._tryShowSubMenu(aEl, op, true);
450 	    	});
451 	    },

481 	    	var eventFunc = $.trim(clickEvent || "");
482 	    	if (eventFunc.length == 0) {
483 	    		return;
484 	    	}
485     		var func = null; 
486 	    	try {
487 	    		func = eval(eventFunc);
488 	    	} catch (e) {

497 			var hrefStr = $.trim(aEl.attr("href") || "");
498 			if (hrefStr.length != 0) {
499 				//{ 20121105 hanyin 修复IE7下，href被自动加上"http://XXX:xxxx/..."的问题
500 				var subMenuId = hrefStr.lastIndexOf("#");
501 				if (subMenuId == -1 && subMenuI+1 >= hrefStr.length) {
502 					return;
503 				}
504 				subMenuId = hrefStr.substring(subMenuId+1);
505 				if (subMenuId.length == 0) {
506 					return false;
507 				}
508 				var subMenuEl = $I(subMenuId);
509 				if (subMenuEl.size() == 0) { // 没有对应的子菜单
510 					return false;
511 				}
512 				//} 20121105 hanyin
513 				// 延时加载下级菜单
514 				var self = this;
515 				var currentActiveId = aEl.attr("id") || "";
516 				var ops = op;
517 				
518 				 var showSubMenu = function() {
519 					 if (self.options._idActiveSubMenu) {
520 						 return; // 子菜单已经显示
521 					 }
522 					// 只有在等待500毫秒之后，焦点还在aEl上，才弹出菜单
523 					var currentFocusId = null;
524 					if (self.options._objFocusItem != null) {
525 						currentFocusId = self.options._objFocusItem.attr("id");
526 					}
527 					if (currentFocusId != currentActiveId) { // 如果焦点已经移开则不弹出
528 						return;
529 					}
530 					// 准备弹出子菜单保存状态
531 					self.options._objActiveItem = aEl;
532 					self.options._idActiveSubMenu = subMenuId;
533 					aEl.addClass("f-state-active").removeClass("f-state-focus");
534 					// 打开下级菜单
535 					var initOption = {};
536 					initOption._parentMenuId = self.options.id; // 传ID比传对象好，减少内存泄露的可能
537 					initOption.attach = aEl;
538 					initOption.direction = "all";
539 					// initOption.offset = ops.offset;
540 					initOption.container = ops.container;
541 					initOption.collision = ops.collision;
542 					subMenuEl.FMenu("show", initOption);
543 				}
544 				
545 				if (immediate === true) {
546 					showSubMenu.call(this);
547 				} else {
548 					setTimeout(showSubMenu, 300);
549 				}
550 			}
