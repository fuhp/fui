 52     $.widget('FUI.FPagingbar', {
 53         options : {
 54             /**
 55              * 每页显示数据的条数。默认值为：10。
 56              * @name FPagingbar#pageSize
 57              * @type Number
 58              * @default 10
 59              * @example
 60              * 无
 61              */
 62             pageSize : 10,
 63             /**
 64              * 每页最多显示的行数，如果设置的pageSize超过此值则会设置为maxPageSize
 65              * @name FPagingbar#maxPageSize
 66              * @type Number
 67              * @default 300
 68              * @example
 69              * 无
 70              */
 71             maxPageSize : 300
 72         },
 73         _create : function() {
 74             var element = this.element;
 75             this.id = this.element.attr('id');
 76             this.start = 1;
 77             this.pageIndex = 1;
 78         },
 79         _init : function() {
 80             var id = this.element.attr('id');
 81             this.firstBtn = $I(id + '-page-first');
 82             this.prevBtn = $I(id + '-page-prev');
 83             this.nextBtn = $I(id + '-page-next');
 84             this.lastBtn = $I(id + '-page-last');
 85             this.refreshBtn = $I(id + '-page-refresh');
 86 
 87             this.beforePageTextSpan = $I(id + "-page-beforeText");
 88             this.afterPageTextSpan = $I(id + "-page-afterText");
 89             this.beforePageSizeTextSpan = $I(id + "-page-size-beforeText");
 90             this.afterPageSizeTextSpan = $I(id + "-page-size-afterText");
 91             
 92             // 当前页行数的输入框
 93             this.pageSizeInput = $I(id+"-page-size-input");
 94 
 95             //当前页输入框
 96             this.inputBtn = $I(id + '-page-input');
 97 
 98             //错误提示信息
 99             this.msgText = $I(id + '-page-totalcount');
100             
101             // 校验pageSize的合法性
102             if (this.beforePageTextSpan.length > 0) {
103 	            this.options.pageSize = parseInt(this.options.pageSize) || 10;
104 	            this.options.maxPageSize = parseInt(this.options.maxPageSize) || 300;
105 	            if (this.options.maxPageSize < 0) {
106 	            	this.options.maxPageSize = 300;
107 	            }
108 	            if (this.options.pageSize < 1) {
109 	            	this.options.pageSize = 1;
110 	            }
111 	            if (this.options.pageSize > this.options.maxPageSize) {
112 	            	this.options.pageSize = this.options.maxPageSize;
113 	            }
114 	            this.pageSizeInput.val(this.options.pageSize);
115             }
116 
117             this._initI18n();
118             this._initMsg();
119             this._bindPageEvent();
120         },
121 
122 
123         //获取ajax请求需要的参数，例如start，limit参数
124         getParams : function() {
125             var params = {};
126             var pageSize = this.options.pageSize;
127             var pageIndex = this.pageIndex;
128             var start = 1 + ((pageIndex - 1) * pageSize);
129             this.start = start;
130             params["start"] = start;
131             params["limit"] = pageSize;
132             params["pageNum"] =  pageIndex ;
133 
134             return params;
135         },
136         //获取组件的默认参数，即start为1，limit为pageSize。
137         getDefaultParams : function() {
138             var params = {};
139             var pageSize = this.options.pageSize;
140             //重置pageIndex和start属性，使调用resetPagebar方法的时候重新计算相关提示信息。
141             this.pageIndex = 1;
142             this.start = 1;
143             params["start"] = this.start;
144             params["limit"] = pageSize;
145             params["pageNum"] =  this.pageIndex ;
146 
147             return params;
148         },
149 
150 
151         //重置分页栏的按钮以及提示信息。
152         resetPagebar : function(listCount, totalCount) {
153             var UTILS = window['$Utils'];
154             var pageIndex = this.pageIndex;
155             this.totalCount = totalCount;
156             var pageSize = this.options.pageSize;
157             this.pageCount = parseInt(this.totalCount / pageSize);
158             if (totalCount % pageSize > 0) {
159                 this.pageCount ++;
160             }
161 
162             if (totalCount) {
163                 //有数据
164                 if (pageIndex < 2) {
165                     this._disableButton(this.firstBtn);
166                     this._disableButton(this.prevBtn);
167                 } else {
168                     this._enableButton(this.firstBtn);
169                     this._enableButton(this.prevBtn);
170                 }
171                 if (pageIndex < this.pageCount) {
172                     this._enableButton(this.nextBtn);
173                     this._enableButton(this.lastBtn);
174                 } else {
175                     this._disableButton(this.nextBtn);
176                     this._disableButton(this.lastBtn);
177                 }
178             } else {
179                 //第一页没有数据
180                 this._disableButton(this.firstBtn);
181                 this._disableButton(this.prevBtn);
182                 this._disableButton(this.nextBtn);
183                 this._disableButton(this.lastBtn);
184             }
185             this._enableButton(this.refreshBtn);
186 
187 
188             //更新分页栏显示信息
189             var start = this.start;
190             var end = start + pageSize - 1;
191             (end > totalCount) && (end = totalCount);
192 
193             //更新当前页码
194             this.inputBtn.val(pageIndex);
195             //更新输入框的提示信息
196             this.pageSizeInput.val(pageSize);
197 
198             this.afterPageTextSpan.text(UTILS.format(this.afterPageText, this.pageCount));
199             this.msgText.text(UTILS.format(this.pageTotalMsg, start, end, totalCount));
200         },
201         _initI18n : function() {
202             var lang = $.FUI.lang && $.FUI.lang.pagingbar || {};
203             this.beforePageText = lang.beforePageText || "第";
204             this.afterPageText = lang.afterPageText || "页 共{0} 页";
205             this.pageTotalMsg = lang.pageTotalMsg || "显示{0}到{1},共{2}条数据";
206             this.beforePageSizeText = lang.beforePageSizeText || "每页";
207             this.afterPageSizeText = lang.afterPageSizeText || "条";
208         },
209         _initMsg : function() {
210             var UTILS = window['$Utils'];
211             this.beforePageTextSpan.text(this.beforePageText);
212             this.afterPageTextSpan.text(UTILS.format(this.afterPageText, "0"));
213             this.msgText.text(UTILS.format(this.pageTotalMsg, "0", "0", "0"));
214             this.beforePageSizeTextSpan.text(this.beforePageSizeText);
215             this.afterPageSizeTextSpan.text(this.afterPageSizeText);
216         },
217 
218         _first : function() {
219             this.pageIndex = 1;
220             this._doLoad();
221         },
222         _prev : function() {
223             var pageIndex = this.pageIndex - 1;
224             pageIndex = pageIndex > 0 ? pageIndex : 1;
225             this.pageIndex = pageIndex;
226             this._doLoad();
227         },
228         _next : function() {
229             this.pageIndex ++;
230             this._doLoad();
231         },
232         _last : function() {
233             this.pageIndex = this.pageCount;
234             this._doLoad();
235         },
236         _refresh : function() {
237         	this._tryPageSizeChange();
238             this._doLoad(true);
239         },
240         
241         _tryPageSizeChange : function() {
242         	// 没有设置pageSize的接口
243         	if (this.pageSizeInput.length == 0) {
244         		return;
245         	}
246             var val = parseInt(this.pageSizeInput.val());
247             if (isNaN(val)) {
248             	this.pageSizeInput.val(this.options.pageSize);
249             	return;
250             }
251             if (val < 1) {
252             	val = 1;
253             }
254             if (val > this.options.maxPageSize) {
255             	val = this.options.maxPageSize;
256             }
257             this.pageSizeInput.val(val);
258             if (this.options.pageSize === val) {
259             	return;
260             }
261             this.options.pageSize = val;
262             
263             this.start = 1;
264             this.pageIndex = 1;
265         },
266 
267         _doLoad : function(isRefresh) {
268             var $C = window['$Component'];
269             if (!this.gridId) {
270                 this.gridId = this.element.parent().attr('id');
271             }
272             //var gridId = this.options.gridId;
273             var params = this.getParams();
274             var gridEl = $I(this.gridId);
275             if (isRefresh) {
276                 //刷新页面，且为跨页选中，那么将当前页选中的内容从crossPageDataCache属性中移除。
277                 $C.tryCall(gridEl, 'resetDataCache', false, isRefresh);
278             } else {
279                 $C.tryCall(gridEl, 'resetDataCache', false);
280             }
281 
282             $C.tryCall(gridEl, 'loadForPagingbar', params);
283             gridEl = null;
284         },
285 
286         //禁用按钮
287         _disableButton : function(btn) {
288             var status = btn.attr("f_grid_page_status");
289             if ("enable" == status) {
290                 var type = btn.attr("f_grid_page_type");
291                 btn.get(0).className = "f-grid-page-button  f-grid-page-" + type + "-disabled";
292                 btn.attr("f_grid_page_status", "disabled");
293             }
294         },
295         //启用按钮
296         _enableButton : function (btn) {
297             var status = btn.attr("f_grid_page_status");
298             if ("disabled" == status) {
299                 var type = btn.attr("f_grid_page_type");
300                 btn.get(0).className = "f-grid-page-button f-grid-page-btn f-grid-page-" + type;
301                 btn.attr("f_grid_page_status", "enable");
302             }
303         },
304         _bindPageEvent : function() {
305             var ME = this;
306             var pageEl = this.element;
307             var inputBtnEl = this.inputBtn;
308             var pageSizeInputEl = this.pageSizeInput;
309             var UTILS = window['$Utils'];
310             pageEl.bind({
311                 click:function(e) {
312                     var tar = e.target;
313                     var type = tar.getAttribute("f_grid_page_type");
314                     var status = tar.getAttribute("f_grid_page_status");
315                     if (type && status !== 'disabled') {
316                         if ("first" == type) {
317                             ME._first();
318                         } else if ("prev" == type) {
319                             ME._prev();
320                         } else if ("next" == type) {
321                             ME._next();
322                         } else if ("last" == type) {
323                             ME._last();
324                         } else if ("refresh" == type) {
325                             ME._refresh();
326                         }
327                     }
328 
329                 },
330                 mouseover : function(e) {
331                     var tar = e.target;
332                     var type = tar.getAttribute("f_grid_page_type");
333                     var status = tar.getAttribute("f_grid_page_status");
334                     if (type && status === 'enable') {
335                         UTILS.replaceClass(tar, "f-grid-page-button f-grid-page-btn f-grid-page-" + type + "-hover")
336                     }
337                 },
338                 mouseout : function(e) {
339                     var tar = e.target;
340                     var type = tar.getAttribute("f_grid_page_type");
341                     var status = tar.getAttribute("f_grid_page_status");
342                     if (type && status === 'enable') {
343                         UTILS.replaceClass(tar, "f-grid-page-button f-grid-page-btn f-grid-page-" + type)
344                     }
345                 },
346                 mousedown:function(e) {
347                     var tar = e.target;
348                     var type = tar.getAttribute("f_grid_page_type");
349                     var status = tar.getAttribute("f_grid_page_status");
350                     if (type && status === 'enable') {
351                         UTILS.replaceClass(tar, "f-grid-page-button f-grid-page-btn f-grid-page-" + type + "-active");
352                     }
353                 },
354                 mouseup:function(e) {
355                     var tar = e.target;
356                     var type = tar.getAttribute("f_grid_page_type");
357                     var status = tar.getAttribute("f_grid_page_status");
358                     if (type && status === "enable") {
359                         UTILS.replaceClass(tar, "f-grid-page-button f-grid-page-btn f-grid-page-" + type + "-hover");
360                     }
361                 }
362             });
363             
364             pageSizeInputEl.bind({
365                 keydown:function(e) {
366                     var keyCode = e.keyCode;
367                     if (keyCode == 13) {
368                         ME._tryPageSizeChange();
369                         ME._doLoad(true);
370                     }
371                 }
372             });
373 
374             inputBtnEl.bind({
375                 blur: function(e) {
376                     ME.inputBtn.val(ME.pageIndex);
377                 },
378                 keydown:function(e) {
379                     var tar = e.target;
380                     var keyCode = e.keyCode;
381                     if (keyCode == 13) {
382                         var inputValue = parseInt(tar.value);
383                         //2013-01-09  start  bug：3976  modify by qudc   将rgExp属性值由/\d+$/（自然数） 修改成 /^[0-9]*[1-9][0-9]*$/（正整数）
384                         var rgExp = /^[0-9]*[1-9][0-9]*$/;
385                         //2013-01-09  end bug：3976  modify by qudc end
386                         var result = rgExp.exec(inputValue);
387                         if (result) {
388                             /* 2013-01-09  start bug：3976 delete  by qudc 注释掉该代码，因为正则校验修改成整数，所以不会出现inputValue值为0或者负数的情况。
389                             if (inputValue < 1) {
390                                 inputValue = 1;
391                                 //ME.totalSpan.text(inputValue);
392                                 ME.afterPageTextSpan.text(UTILS.format(ME.afterPageText, inputValue));
393                             }
394                               2013-01-09  end  bug：3976 delete by qudc
395                             */
396                             if (inputValue > ME.pageCount) {
397                                 inputValue = ME.pageCount;
398                                 //ME.totalSpan.text(inputValue);
399                                 ME.afterPageTextSpan.text(UTILS.format(ME.afterPageText, inputValue));
400                             }
401                             if (ME.pageIndex != inputValue) {
402                                 ME.pageIndex = inputValue;
403                                 ME._doLoad();
404                             }
405                         } else {
406                             ME.inputBtn.val(ME.pageIndex);
407                         }
408                     }
409                 }
410             })
411 
412         },
413          destroy : function() {
414             this.element.unbind();
415             this.inputBtn.unbind();
416             this.inputBtn = null;
417             this.firstBtn = null;
418             this.prevBtn = null;
419             this.nextBtn = null;
420             this.lastBtn = null;
421             this.refreshBtn = null;
422             this.beforePageTextSpan = null;
423             this.afterPageTextSpan = null;
424             this.beforePageSizeTextSpan = null;
425             this.afterPageSizeTextSpan = null;
426             //错误提示信息
427             this.msgText = null;
428         }
