177 	    	var self = this;
178 	    	var op = self.options;
179 	    	var headers = op._objHeaders;
180 	    	var size = headers.size();
181 	    	var id = headers.element(n).key;
182 	    	if (id == op._curActiveId){ // 已经被激活，则隐藏
183 	    		if (n >=0 && n < size-1) {
184 	    			n = n+1;
185 	    		} else if (n == size-1) { // 已经是最后一个则激活前一个
186 	    			n = n-1
187 	    		}
188 	    	}
189 	    	self.activate(n);

192 	    /**
193 	     * 激活指定的抽屉。index为整数或者抽屉的id。任何其它数据将会用parseInt及isNaN进行处理。 (注意，如果组件为禁用状态，执行此方法无任何效果)
194 	     * <ul>
195 	     * <li>如果抽屉个数为0，则不激活任何抽屉</li>
196 	     * <li>如果抽屉个数大于0，且index<0，则激活第一个抽屉(索引为0的那个抽屉)</li>
197 	     * <li>如果抽屉个数大于0，且index>=抽屉的个数，则激活最后一个抽屉</li>
198 	     * </ul>
199 	     * @name FAccordion#activate
200 	     * @function
201 	     * @param index 要激活的抽屉的索引(从0开始)或者抽屉的id
202 	     * @example
203 	     */
204 	    activate : function(n) {
205 	    	var self = this;
206 	    	var op = self.options;
207 	    	var headers = op._objHeaders;
208 	    	var size = headers.size();
209 	    	
210 	    	var el = self._parseToEl(n, headers);
211 	    	if (!el || el.length == 0) {
212 	    		return false;
213 	    	}
214 	    	
215 	    	var id = el.attr("id");
216 	    	var index = headers.indexOf(id);
217 	    	var curActiveId = op._curActiveId;
218 	    	if (id === curActiveId) { // 如果已经处于开启状态，则不重复展开
219 	    		return false;
220 	    	}
221 	    	
222 	    	// 调用hide回调，如果返回false，则阻止展开
223 	    	if (op._curActiveId && op.onBeforeCollapse) {
224 	    		var result = self._tryExcuteFunc(self, op.onBeforeCollapse, [headers.indexOf(op._curActiveId)]);
225 	    		if (result === false) {
226 	    			return false;
227 	    		}
228 	    	}
229 	    	
230 	    	// 调用show回调，如果返回false，则阻止激活
231 	    	if (op.onBeforeActive) {
232 	    		var result = self._tryExcuteFunc(self, op.onBeforeActive, [index]);
233 	    		if (result === false) {
234 	    			return false;
235 	    		}
236 	    	}
237 	    	
238 	    	// 隐藏当前被激活的页签
239 	    	var toHide = self._hideCurActive();
240 	    	// 显示此时被激活的页签
241 	    	var toShow = self._showActive(index, id, el);
242 	    	// 执行动画
243 //	    	if ($.browser.msie && ($.browser.version == "6.0") && !$.support.style) {
244 	    		!toHide || toHide.addClass("f-state-hide");
245 	    		toShow.removeClass("f-state-hide");
246 //	    	} else {
247 //		    	self._animate(toHide, toShow);
248 //	    	}
249 	    	var lasActiveId = op._curActiveId;
250 		    op._curActiveId = id;
251 	    	
252 	    	!op.onCollapse || self._tryExcuteFunc(self, op.onCollapse, [headers.indexOf(lasActiveId)]);
253 	    	!op.onActive || self._tryExcuteFunc(self, op.onActive, [index]);
254 	    },

256     	// 之所以不直接使用slideUp和slideDown，是为了避免在动画过程中出现抖动的问题
257 	    _animate : function(toHide, toShow) {
258 	    	var op = this.options;
259 	    	var height = op._contentHeight;
260 			var adjust = 0;
261 	    	!toHide || toHide.stop(true,true).animate( op._hideProps, {
262 				duration: "fast",
263 				step : function( now, fx ) {
264 					fx.now = Math.round( now );
265 				}
266 			});
267 			toShow.stop(true,true).animate( op._showProps, {
268 				duration: "fast",
269 				step : function( now, fx ) {
270 					fx.now = Math.round( now );
271 					if (fx.prop !== "height") {
272 						adjust += fx.now;
273 					} else if (toHide && !isNaN(height)) {
274 						fx.now = Math.round( height - toHide.height() - adjust);
275 						adjust = 0;
276 					}
277 				}
278 			});
279 	    },

319 	    _parseToEl : function(n, headers) {
320 	    	if ($Utils.isJQueryObj(n)) { // jquery对象
321 	    		return n;
322 	    	}
323 	    	headers = headers || this.options._objHeaders;
324 	    	var size = headers.size();
325 	    	if (!isNaN(n)) { // 索引
326 	    		if (n < 0) {
327 	    			n = 0;
328 	    		} else if (n >= size) {
329 	    			n = size-1;
330 	    		}
331 	    		return headers.element(n).value;
332 	    	}
333 	    	if ($Utils.isString(n)) { // id
334 	    		return this.element.children("#"+n);
335 	    	}
336 	    	return null;
337 	    },

399 	    /**
400 	     * 设置组件的大小
401 	     * @name FAccordion#setSize
402 	     * @function
403 	     * @param w 组件的宽度，支持Number/String，比如 500或者"500px"，不能传入其他的无效值
404 	     * @param h 组件的高度，支持Number/String，比如 500或者"500px"，不能传入其他的无效值
405 	     * @example
406 	     */
407 	    setSize : function(w, h) {
408 	    	var self = this;
409 	    	var op = self.options;
410 	    	var selfEl = this.element;
411 	    	
412 	    	if (w) { // 设置宽度
413 	    		w = parseInt(w);
414 	    		if (!isNaN(w)) {
415 	    			w = w-2;  // 边框2像素
416 	    			if (w < 0) {w=0;}
417 	    		} else {
418 	    			w = selfEl.parent().width() -2; // 自适应父亲的高度
419 	    			if (w < 0) {w=0;}
420 	    		}
421 	    		if (op.width !== w) {
422 	    			selfEl.width(w);
423 	    			self._adjustWidth(w);
424 	    		}
425     			op.width = w;
426 	    	}
427 	    	if (h) { // 设置高度
428 	    		var rh = parseInt(h);
429 	    		if (!isNaN(rh)) {
430 	    			rh = rh-2; // 边框2像素
431 	    			if (rh < 0) {rh=0;}
432 	    		} else if (h === "fit") {
433 	    			rh = selfEl.parent().height() -2; // 自适应父亲的高度
434 	    			if (rh < 0) {rh=0;}
435 	    		} else {
436 	    			rh = "auto";
437 	    		}
438 	    		if (op.height !== rh) {
439 		    		selfEl.height(rh);
440 		    		// add 20130110 hanyin 在大小变化是，设置content的宽度
441 		    		self._adjustHeight(rh); // 调整内部content的高度
442 		    		// end add 20130110 hanyin 
443 	    		}
444 	    		op.height = h;
445 	    	}

482 	    		contentEl.height(contentHeight);

488 	    // 如果传入的是function，则直接执行，否则认为传入的字符串是一个方法的名字，并执行这个方法
489 	    _tryExcuteFunc : function(obj, func, args) {
490 	    	var EMPTY_FUNC = function() {};
491 	    	var f = func || EMPTY_FUNC;
492 	    	if (!$.isFunction(func)) {
493 	    		try {
494 	    			f = eval(func);
495 	    			if (!$.isFunction(f)) {
496 	    				return;
497 	    			}
498 	    		} catch (e) {
499 	    			return;
500 				}
501 	    	}
502 	    	args = args || [];
503 	    	return f.apply(obj, args);
504 	    }
